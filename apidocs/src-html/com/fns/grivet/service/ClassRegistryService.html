001package com.fns.grivet.service;
002
003import java.time.LocalDateTime;
004import java.util.Iterator;
005import java.util.List;
006import java.util.Set;
007
008import org.apache.commons.lang3.StringUtils;
009import org.json.JSONArray;
010import org.json.JSONObject;
011import org.springframework.beans.factory.annotation.Autowired;
012import org.springframework.dao.DuplicateKeyException;
013import org.springframework.stereotype.Service;
014import org.springframework.transaction.annotation.Transactional;
015import org.springframework.util.Assert;
016
017import com.fns.grivet.model.Attribute;
018import com.fns.grivet.model.AttributeType;
019import com.fns.grivet.model.ClassAttribute;
020import com.fns.grivet.repo.AttributeRepository;
021import com.fns.grivet.repo.AttributeTypeRepository;
022import com.fns.grivet.repo.ClassAttributeRepository;
023import com.fns.grivet.repo.ClassRepository;
024
025@Service
026public class ClassRegistryService {
027
028    private static final String ID = "id";
029    private static final String SCHEMA = "$schema";
030    private static final String TYPE = "type";
031    private static final String DESCRIPTION = "description";
032    private static final String ATTRIBUTES = "attributes";
033    private static final String PROPERTIES = "properties";
034    
035    private final AttributeRepository attributeRepository;
036    private final AttributeTypeRepository attributeTypeRepository;
037    private final ClassRepository classRepository;
038    private final ClassAttributeRepository classAttributeRepository;
039    
040    @Autowired
041    public ClassRegistryService(AttributeRepository attributeRepository,
042            AttributeTypeRepository attributeTypeRepository, ClassRepository classRepository,
043            ClassAttributeRepository classAttributeRepository) {
044        this.attributeRepository = attributeRepository;
045        this.attributeTypeRepository = attributeTypeRepository;
046        this.classRepository = classRepository;
047        this.classAttributeRepository = classAttributeRepository;
048    }
049
050    @Transactional
051    // USE WITH CAUTION! Consider that w/ FK, type and all related data will be destroyed
052    public void deregister(String type) {
053        com.fns.grivet.model.Class c = classRepository.findByName(type);
054        if (c != null) {
055            classRepository.delete(c);
056        }
057    }
058    
059    @Transactional
060    public String register(JSONObject payload) {
061        String type = payload.getString(TYPE);
062        String description = payload.optString(DESCRIPTION);
063        com.fns.grivet.model.Class c = classRepository.findByName(type);
064        if (c == null) {
065            c = classRepository.save(new com.fns.grivet.model.Class(type, description));
066            JSONObject attributes = payload.getJSONObject(ATTRIBUTES);
067            Set<String> attributeNames = attributes.keySet();
068            Assert.notEmpty(attributeNames, String.format("[%s] must declare at least one attribute in registration request!", type));
069            Attribute a = null;
070            String at = null;
071            AttributeType attributeType = null;
072            // TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
073            for (String attributeName: attributeNames) {
074                a = attributeRepository.findByName(attributeName);
075                if (a == null) {
076                    a = attributeRepository.save(new Attribute(attributeName));
077                }
078                at = attributes.getString(attributeName);
079                Assert.notNull(at, String.format("[%s].[%s] must declare an attribute type in registration request!", type, attributeName));
080                attributeType = attributeTypeRepository.findByType(at);
081                Assert.notNull(attributeType, String.format("Attribute type [%s] is not supported!", at));
082                classAttributeRepository.save(new ClassAttribute(c.getId(), a.getId(), attributeType.getId()));
083            }
084        } else {
085            throw new DuplicateKeyException(String.format("Type [%s] already registered!", type));
086        }
087        return type;
088    }
089
090    @Transactional(readOnly=true)
091    public JSONObject get(String type) {
092        com.fns.grivet.model.Class c = classRepository.findByName(type);
093        Assert.notNull(c, String.format("Type [%s] is not registered!", type));
094        JSONObject result = new JSONObject();
095        result.put("type", type);
096        String description = c.getDescription();
097        result.put("description", description != null ? description: "");
098        List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
099        Assert.notEmpty(cas, String.format("Type [%s] does not have any attributes registered!", type));
100        JSONObject attributes = new JSONObject();
101        result.put("attributes", attributes);
102        Attribute a = null;
103        AttributeType at = null;
104        for (ClassAttribute ca: cas) {
105            a = attributeRepository.findOne(ca.getAid());
106            Assert.notNull(a, String.format("Attribute id [%s] is not registered!", ca.getAid()));
107            at = attributeTypeRepository.findOne(ca.getTid());
108            Assert.notNull(a, String.format("Attribute Type id [%s] is not registered!", ca.getTid()));
109            attributes.put(a.getName(), at.getType());
110        }
111        return result;
112    }
113    
114    @Transactional(readOnly=true)
115    public JSONArray all() {
116        JSONArray result = new JSONArray();
117        Iterable<com.fns.grivet.model.Class> iterable = classRepository.findAll();
118        Assert.notNull(iterable, "No types are registered!");
119        Iterator<com.fns.grivet.model.Class> it = iterable.iterator();
120        com.fns.grivet.model.Class c = null;
121        while(it.hasNext()) {
122           c = it.next();
123           result.put(get(c.getName()));
124        }
125        return result;
126    }
127    
128    public boolean isJsonSchema(JSONObject payload) {
129        boolean result = false;
130        String schema = payload.optString(SCHEMA);
131        String type = payload.optString(TYPE);
132        String id = payload.optString(ID);
133        String props = payload.optString(PROPERTIES);
134        if (schema.equals("http://json-schema.org/draft-04/schema#") 
135                && type.equals("object") 
136                && StringUtils.isNotBlank(id) 
137                && StringUtils.isNotBlank(props)) {
138            result = true;
139        }
140        return result;
141    }
142
143    @Transactional
144    public com.fns.grivet.model.Class linkSchema(JSONObject payload) {
145        String type = payload.getString(ID);
146        com.fns.grivet.model.Class c = classRepository.findByName(type);
147        Assert.notNull(c, String.format("Type [%s] must be registered before linking a JSON Schema!", type));
148        c.setValidatable(true);
149        c.setJsonSchema(payload.toString());
150        c.setUpdatedTime(LocalDateTime.now());
151        classRepository.save(c);
152        return c;
153    }
154    
155    @Transactional
156    public com.fns.grivet.model.Class unlinkSchema(String type) {
157        com.fns.grivet.model.Class c = classRepository.findByName(type);
158        Assert.notNull(c, String.format("Type [%s] must be registered before unlinking a JSON Schema!", type));
159        c.setValidatable(false);
160        c.setJsonSchema(null);
161        c.setUpdatedTime(LocalDateTime.now());
162        classRepository.save(c);
163        return c;
164    }
165    
166}
