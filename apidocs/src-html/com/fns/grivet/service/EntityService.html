033@Service
034public class EntityService {
035
036    private final ClassRepository classRepository;
037    private final ClassAttributeRepository classAttributeRepository;
038    private final AttributeRepository attributeRepository;
039    private final AttributeTypeRepository attributeTypeRepository;
040    private final EntityRepository entityRepository;
041    private final SchemaValidator schemaValidator;
042    
043    @Autowired
044    public EntityService(ClassRepository classRepository, ClassAttributeRepository classAttributeRepository,
045            AttributeRepository attributeRepository, AttributeTypeRepository attributeTypeRepository,
046            EntityRepository entityRepository, SchemaValidator schemaValidator) {
047        this.classRepository = classRepository;
048        this.classAttributeRepository = classAttributeRepository;
049        this.attributeRepository = attributeRepository;
050        this.attributeTypeRepository = attributeTypeRepository;
051        this.entityRepository = entityRepository;
052        this.schemaValidator = schemaValidator;
053    }
054
055    @Transactional
056    public void create(String type, JSONObject payload) {
057        com.fns.grivet.model.Class c = classRepository.findByName(type);
058        Assert.notNull(c, String.format("Type [%s] is not registered!", type));
059        if (c.isValidatable()) {
060            ProcessingReport report = schemaValidator.validate(type, payload);
061            Assert.isTrue(report.isSuccess(), String.format("Cannot store [%s]! Type does not conform to JSON Schema definition.\n %s", type, report.toString()));
062        }
063        Set<String> keys = payload.keySet();
064        Assert.notEmpty(keys, String.format("Type [%s] must declare at least one attribute in create request!", type));
065        Object val = null;
066        Attribute a = null;
067        ClassAttribute ca = null;
068        AttributeType at = null;
069        Long eid = entityRepository.id(c.getId());
070        // TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
071        for (String key: keys) {
072            val = payload.get(key);
073            a = attributeRepository.findByName(key);
074            Assert.notNull(a, String.format("Attribute [%s] is not registered!", key));
075            ca = classAttributeRepository.findByCidAndAid(c.getId(), a.getId());
076            Assert.notNull(ca, String.format("[%s] is not a valid attribute of [%s]", key, type));
077            at = attributeTypeRepository.findOne(ca.getTid());
078            Assert.notNull(at, String.format("Attribute type [%s] is not supported!", at));
079            entityRepository.save(eid, a, at, val);
080        }
081    }
082    
083    @Transactional(readOnly=true)
084    public String get(String type, LocalDateTime createdTimeStart, LocalDateTime createdTimeEnd, Set<Entry<String, String[]>> requestParameters) throws JsonProcessingException {
085        com.fns.grivet.model.Class c = classRepository.findByName(type);
086        Assert.notNull(c, "Type [%s] is not registered!");
087        List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
088        List<Attribute> attributes = Lists.newArrayList(attributeRepository.findAll());
089        Map<Integer, Integer> attributeToAttributeTypeMap = cas.stream().collect(Collectors.toMap(ClassAttribute::getAid, ClassAttribute::getTid));
090        Map<String, Integer> attributeNameToAttributeIdMap = attributes.stream().collect(Collectors.toMap(Attribute::getName, Attribute::getId));
091        DynamicQuery query = new DynamicQuery(requestParameters, attributeToAttributeTypeMap, attributeNameToAttributeIdMap);
092        List<EntityAttributeValue> rows = null;
093        if (query.hasConstraints()) {
094            rows = entityRepository.executeDynamicQuery(c.getId(), query);
095        } else {
096            rows = entityRepository.find(c.getId(), createdTimeStart, createdTimeEnd);
097        }
098        return mapRows(attributeToAttributeTypeMap, rows);
099    }
100
101    protected String mapRows(Map<Integer, Integer> attributeToAttributeTypeMap, List<EntityAttributeValue> rows) {
102        JSONArray jsonArray = new JSONArray();
103        Long current = null; 
104        Long previous = null;
105        JSONObject jsonObject = null;
106        for (EntityAttributeValue row: rows) {
107            current = row.getId();
108            if (current != previous) {
109                jsonObject = new JSONObject();
110                jsonArray.put(jsonObject);
111            }
112            Integer tid = attributeToAttributeTypeMap.get(row.getAttributeId());
113            jsonObject.put(row.getAttributeName(), ValueHelper.getValue(attributeTypeRepository.findOne(tid), row.getAttributeValue()));
114            previous = current;
115        }
116        return jsonArray.toString();
117    }
118}
