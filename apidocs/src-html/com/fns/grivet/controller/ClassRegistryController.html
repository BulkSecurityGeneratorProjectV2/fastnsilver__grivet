001package com.fns.grivet.controller;
002
003import java.io.IOException;
004import java.net.URI;
005
006import javax.servlet.http.HttpServletRequest;
007
008import org.apache.commons.io.IOUtils;
009import org.json.JSONArray;
010import org.json.JSONObject;
011import org.springframework.beans.factory.annotation.Autowired;
012import org.springframework.http.HttpHeaders;
013import org.springframework.http.HttpStatus;
014import org.springframework.http.MediaType;
015import org.springframework.http.ResponseEntity;
016import org.springframework.util.Assert;
017import org.springframework.web.bind.annotation.PathVariable;
018import org.springframework.web.bind.annotation.RequestMapping;
019import org.springframework.web.bind.annotation.RequestMethod;
020import org.springframework.web.bind.annotation.RestController;
021import org.springframework.web.util.UriComponentsBuilder;
022
023import com.fns.grivet.service.ClassRegistryService;
024
025
026@RestController
027@RequestMapping("/register")
028public class ClassRegistryController {
029
030    private final ClassRegistryService classRegistryService;
031    
032    @Autowired
033    public ClassRegistryController(ClassRegistryService classRegistryService) {
034        this.classRegistryService = classRegistryService;
035    }
036    
037    @RequestMapping(method=RequestMethod.POST, 
038            consumes=MediaType.APPLICATION_JSON_VALUE, produces=MediaType.APPLICATION_JSON_VALUE)
039    public ResponseEntity<?> register(HttpServletRequest request) throws IOException {
040        String json = IOUtils.toString(request.getInputStream(), "UTF-8");
041        Assert.isTrue(json.startsWith("{") || json.startsWith("["), "Registration requests must be valid JSON starting with either a { or [!");
042        ResponseEntity<?> result = ResponseEntity.unprocessableEntity().build();
043        if (json.startsWith("{")) {
044            if (request.getParameterMap().containsKey("linkSchema") 
045                    && classRegistryService.isJsonSchema(new JSONObject(json))) {
046                result = linkSchema(json);
047            } else {
048                result = registerSingleType(json);
049            }
050        }
051        if (json.startsWith("[")) {
052            result = registerMultipleTypes(json);
053        }
054        return result;
055    }
056    
057    private ResponseEntity<?> linkSchema(String json) {
058        String id = classRegistryService.linkSchema(new JSONObject(json)).getName();
059        return ResponseEntity.ok(String.format("JSON Schema for type [%s] linked!  Store requests for this type will be validated henceforth!", id));
060    }
061    
062    private ResponseEntity<?> unlinkSchema(String type) {
063        classRegistryService.unlinkSchema(type);
064        return ResponseEntity.ok(String.format("JSON Schema for type [%s] unlinked!  Store requests for this type will no longer be validated!", type));
065    }
066    
067    private ResponseEntity<?> registerSingleType(String json) {
068        String type = classRegistryService.register(new JSONObject(json));
069        UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
070        return ResponseEntity.created(ucb.path("/register/{type}").buildAndExpand(type).toUri()).build();
071    }
072    
073    private ResponseEntity<?> registerMultipleTypes(String json) {
074        JSONArray jsonArray = new JSONArray(json);
075        JSONObject jsonObject = null;
076        String type = null;
077        HttpHeaders headers = new HttpHeaders();
078        UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
079        int numberOfTypesToRegister = jsonArray.length();
080        URI location = null;
081        int errorCount = 0;
082        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
083        for (int i = 0; i < numberOfTypesToRegister; i++) {
084            jsonObject = jsonArray.getJSONObject(i);
085            try {
086                type = classRegistryService.register(jsonObject);
087                location = ucb.path("/register/{type}").buildAndExpand(type).toUri();
088                if (numberOfTypesToRegister == 1) {
089                    headers.setLocation(location); 
090                } else {
091                    headers.set(String.format("Location[%s]",String.valueOf(i+1)), location.toASCIIString());
092                }
093            } catch (Exception e) {
094                if (numberOfTypesToRegister == 1) {
095                    throw e;
096                }
097                headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
098                errorCount++;
099            }
100        }
101        HttpStatus status = (errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED;
102        return new ResponseEntity<>(headers, status);
103    }
104    
105    @RequestMapping(value="/{type}", produces=MediaType.APPLICATION_JSON_VALUE)
106    public ResponseEntity<?> get(@PathVariable("type") String type) {
107        JSONObject payload = classRegistryService.get(type);
108        return ResponseEntity.ok(payload.toString());
109    }
110    
111    @RequestMapping(value="/{type}", method=RequestMethod.PUT, produces=MediaType.APPLICATION_JSON_VALUE)
112    public ResponseEntity<?> unlinkSchema(@PathVariable("type") String type, HttpServletRequest request) {
113        Assert.isTrue(request.getParameterMap().containsKey("unlinkSchema"), "Operation not supported!"); 
114        return unlinkSchema(type);
115    }
116    
117    @RequestMapping(value="/", produces=MediaType.APPLICATION_JSON_VALUE)
118    public ResponseEntity<?> all() {
119        JSONArray payload = classRegistryService.all();
120        return ResponseEntity.ok(payload.toString());
121    }
122        
123}
