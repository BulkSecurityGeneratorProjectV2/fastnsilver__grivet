016package com.fns.grivet.controller;
017
018import java.io.IOException;
019import java.net.URI;
020
021import javax.servlet.http.HttpServletRequest;
022
023import org.apache.commons.io.IOUtils;
024import org.json.JSONArray;
025import org.json.JSONObject;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028import org.springframework.beans.factory.annotation.Autowired;
029import org.springframework.http.HttpHeaders;
030import org.springframework.http.HttpStatus;
031import org.springframework.http.MediaType;
032import org.springframework.http.ResponseEntity;
033import org.springframework.util.Assert;
034import org.springframework.web.bind.annotation.PathVariable;
035import org.springframework.web.bind.annotation.RequestMapping;
036import org.springframework.web.bind.annotation.RequestMethod;
037import org.springframework.web.bind.annotation.RestController;
038import org.springframework.web.util.UriComponentsBuilder;
039
040import com.fns.grivet.service.ClassRegistryService;
041
042
043/**
044 * Provides end-points for type registration and verification
045 * 
046 * @author Chris Phillipson
047 */
048@RestController
049@RequestMapping("/register")
050public class ClassRegistryController {
051
052    private final Logger log = LoggerFactory.getLogger(getClass());
053    
054    private final ClassRegistryService classRegistryService;
055    
056    @Autowired
057    public ClassRegistryController(ClassRegistryService classRegistryService) {
058        this.classRegistryService = classRegistryService;
059    }
060    
061    @RequestMapping(method=RequestMethod.POST, 
062            consumes=MediaType.APPLICATION_JSON_VALUE, produces=MediaType.APPLICATION_JSON_VALUE)
063    public ResponseEntity<?> register(HttpServletRequest request) throws IOException {
064        String json = IOUtils.toString(request.getInputStream(), "UTF-8");
065        Assert.isTrue(json.startsWith("{") || json.startsWith("["), "Registration requests must be valid JSON starting with either a { or [!");
066        ResponseEntity<?> result = ResponseEntity.unprocessableEntity().build();
067        if (json.startsWith("{")) {
068            if (request.getParameterMap().containsKey("linkSchema") 
069                    && classRegistryService.isJsonSchema(new JSONObject(json))) {
070                result = linkSchema(json);
071            } else {
072                result = registerSingleType(json);
073            }
074        }
075        if (json.startsWith("[")) {
076            result = registerMultipleTypes(json);
077        }
078        return result;
079    }
080    
081    private ResponseEntity<?> linkSchema(String json) {
082        String id = classRegistryService.linkSchema(new JSONObject(json)).getName();
083        String message = String.format("JSON Schema for type [%s] linked!  Store requests for this type will be validated henceforth!", id);
084        log.info(message);
085        return ResponseEntity.ok(message);
086    }
087    
088    private ResponseEntity<?> unlinkSchema(String type) {
089        classRegistryService.unlinkSchema(type);
090        String message = String.format("JSON Schema for type [%s] unlinked!  Store requests for this type will no longer be validated!", type);
091        log.info(message);
092        return ResponseEntity.ok(message);
093    }
094    
095    private ResponseEntity<?> registerSingleType(String json) {
096        String type = classRegistryService.register(new JSONObject(json));
097        UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
098        log.info("Type [{}] successfully registered!", type);
099        return ResponseEntity.created(ucb.path("/register/{type}").buildAndExpand(type).toUri()).build();
100    }
101    
102    private ResponseEntity<?> registerMultipleTypes(String json) {
103        JSONArray jsonArray = new JSONArray(json);
104        JSONObject jsonObject = null;
105        String type = null;
106        HttpHeaders headers = new HttpHeaders();
107        UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
108        int numberOfTypesToRegister = jsonArray.length();
109        URI location = null;
110        int errorCount = 0;
111        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
112        for (int i = 0; i < numberOfTypesToRegister; i++) {
113            jsonObject = jsonArray.getJSONObject(i);
114            try {
115                type = classRegistryService.register(jsonObject);
116                location = ucb.path("/register/{type}").buildAndExpand(type).toUri();
117                if (numberOfTypesToRegister == 1) {
118                    headers.setLocation(location); 
119                } else {
120                    headers.set(String.format("Location[%s]",String.valueOf(i+1)), location.toASCIIString());
121                }
122                log.info("Type [{}] successfully registered!", type);
123            } catch (Exception e) {
124                String message = LogUtil.toLog(jsonObject, String.format("Problems registering type! Portion of payload @ index[%d]\n", i+1));
125                log.error(message, e);
126                if (numberOfTypesToRegister == 1) {
127                    throw e;
128                }
129                headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
130                errorCount++;
131            }
132        }
133        HttpStatus status = (errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED;
134        return new ResponseEntity<>(headers, status);
135    }
136    
137    @RequestMapping(value="/{type}", method=RequestMethod.DELETE, produces=MediaType.APPLICATION_JSON_VALUE)
138    public ResponseEntity<?> delete(@PathVariable("type") String type) {
139        classRegistryService.deregister(type);
140        log.info("Type [{}] successfully deregistered!", type);
141        return ResponseEntity.noContent().build();
142    }
143    
144    @RequestMapping(value="/{type}", produces=MediaType.APPLICATION_JSON_VALUE)
145    public ResponseEntity<?> get(@PathVariable("type") String type) {
146        JSONObject payload = classRegistryService.get(type);
147        String message = LogUtil.toLog(payload, String.format("Successfully retrieved type [%s]\n", type));
148        log.info(message);
149        return ResponseEntity.ok(payload.toString());
150    }
151    
152    @RequestMapping(value="/{type}", method=RequestMethod.PUT, produces=MediaType.APPLICATION_JSON_VALUE)
153    public ResponseEntity<?> unlinkSchema(@PathVariable("type") String type, HttpServletRequest request) {
154        Assert.isTrue(request.getParameterMap().containsKey("unlinkSchema"), "Operation not supported!"); 
155        return unlinkSchema(type);
156    }
157    
158    @RequestMapping(value="/", produces=MediaType.APPLICATION_JSON_VALUE)
159    public ResponseEntity<?> all() {
160        JSONArray payload = classRegistryService.all();
161        return ResponseEntity.ok(payload.toString());
162    }
163        
164}
