001package com.fns.grivet.controller;
002
003import java.io.IOException;
004import java.time.LocalDateTime;
005import java.time.temporal.ChronoUnit;
006
007import javax.servlet.http.HttpServletRequest;
008
009import org.apache.commons.io.IOUtils;
010import org.json.JSONArray;
011import org.json.JSONObject;
012import org.springframework.beans.factory.annotation.Autowired;
013import org.springframework.http.HttpHeaders;
014import org.springframework.http.HttpStatus;
015import org.springframework.http.MediaType;
016import org.springframework.http.ResponseEntity;
017import org.springframework.util.Assert;
018import org.springframework.web.bind.annotation.PathVariable;
019import org.springframework.web.bind.annotation.RequestMapping;
020import org.springframework.web.bind.annotation.RequestMethod;
021import org.springframework.web.bind.annotation.RequestParam;
022import org.springframework.web.bind.annotation.RestController;
023
024import com.codahale.metrics.MetricRegistry;
025import com.fasterxml.jackson.core.JsonProcessingException;
026import com.fns.grivet.service.EntityService;
027
028
029@RestController
030@RequestMapping("/store")
031public class EntityController {
032
033    private final EntityService entityService;
034    private MetricRegistry metricRegistry;
035    
036    @Autowired
037    public EntityController(EntityService entityService) {
038        this.entityService = entityService;
039    }
040    
041    @Autowired
042    public void setMetricRegistry(MetricRegistry metricRegistry) {
043        this.metricRegistry = metricRegistry;
044    }
045
046    @RequestMapping(value="/{type}", method=RequestMethod.POST, 
047            consumes=MediaType.APPLICATION_JSON_VALUE, produces=MediaType.APPLICATION_JSON_VALUE)
048    public ResponseEntity<?> create(@PathVariable("type") String type, HttpServletRequest request) throws IOException {
049        String json = IOUtils.toString(request.getInputStream(), "UTF-8");
050        Assert.isTrue(json.startsWith("{") || json.startsWith("["), "Store requests must be valid JSON starting with either a { or [!");
051        ResponseEntity<?> result = ResponseEntity.unprocessableEntity().build();
052        if (json.startsWith("{")) {
053            result = createSingleType(type, json);
054        }
055        if (json.startsWith("[")) {
056            result = createMultipleTypes(type, json);
057        }
058        return result;
059    }
060    
061    private ResponseEntity<?> createSingleType(String type, String json) {
062        entityService.create(type, new JSONObject(json));
063        metricRegistry.counter(MetricRegistry.name("store", type, "count")).inc();
064        return ResponseEntity.noContent().build();
065    }
066    
067    private ResponseEntity<?> createMultipleTypes(String type, String json) {
068        JSONArray jsonArray = new JSONArray(json);
069        int numberOfTypesToCreate = jsonArray.length();
070        int errorCount = 0;
071        JSONObject jsonObject = null;
072        HttpHeaders headers = new HttpHeaders();
073        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
074        for (int i = 0; i < numberOfTypesToCreate; i++) {
075            jsonObject = jsonArray.getJSONObject(i);
076            try {
077                entityService.create(type, jsonObject);
078                metricRegistry.counter(MetricRegistry.name("store", type, "count")).inc();
079            } catch (Exception e) {
080                if (numberOfTypesToCreate == 1) {
081                    throw e;
082                }
083                headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
084                errorCount++;
085            }
086        }
087        HttpStatus status = (errorCount == 0) ? HttpStatus.NO_CONTENT : HttpStatus.ACCEPTED;
088        return new ResponseEntity<>(headers, status);
089    }
090    
091    @RequestMapping(value="/{type}", produces=MediaType.APPLICATION_JSON_VALUE)
092    public ResponseEntity<?> get(@PathVariable("type") String type, @RequestParam(value="createdTimeStart", required=false) String createdTimeStart, @RequestParam(value="createdTimeEnd", required=false) String createdTimeEnd, HttpServletRequest request) throws JsonProcessingException {
093        LocalDateTime start = createdTimeStart == null ? LocalDateTime.now().minusDays(7): LocalDateTime.parse(createdTimeStart);
094        LocalDateTime end = createdTimeEnd == null ? LocalDateTime.now() : LocalDateTime.parse(createdTimeEnd);
095        Assert.isTrue(ChronoUnit.SECONDS.between(start, end) >= 0, "Store request constraint createdTimeStart must be earlier or equal to createdTimeEnd!");
096        return ResponseEntity.ok(entityService.get(type, start, end, request.getParameterMap().entrySet()));
097    }
098    
099}
