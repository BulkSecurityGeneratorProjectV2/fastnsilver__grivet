001package com.fns.grivet.controller;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonProperty;
005import com.fasterxml.jackson.annotation.JsonPropertyOrder;
006import com.stormpath.sdk.account.Account;
007import com.stormpath.sdk.servlet.account.AccountResolver;
008
009import org.springframework.context.annotation.Profile;
010import org.springframework.http.MediaType;
011import org.springframework.web.bind.annotation.RequestMapping;
012import org.springframework.web.bind.annotation.RestController;
013
014import javax.servlet.http.HttpServletRequest;
015
016@Profile("secure")
017@RestController
018public class MeController {
019
020    @RequestMapping(value = "/me", produces = MediaType.APPLICATION_JSON_VALUE)
021    public AccountInfo info(HttpServletRequest req) {
022        Account account = AccountResolver.INSTANCE.getAccount(req);
023        return new AccountInfo(account.getHref(), account.getFullName(), account.getEmail());
024    }
025
026    @JsonPropertyOrder(value = { "href", "fullName", "email " })
027    class AccountInfo {
028
029        private String href;
030        private String fullName;
031        private String email;
032
033        @JsonCreator
034        public AccountInfo(@JsonProperty String href, @JsonProperty String fullName, @JsonProperty String email) {
035            this.href = href;
036            this.fullName = fullName;
037            this.email = email;
038        }
039
040        public String getHref() {
041            return href;
042        }
043
044        public String getFullName() {
045            return fullName;
046        }
047
048        public String getEmail() {
049            return email;
050        }
051
052    }
053}
