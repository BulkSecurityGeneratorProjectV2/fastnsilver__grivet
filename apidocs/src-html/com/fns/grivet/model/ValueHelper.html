001package com.fns.grivet.model;
002
003import java.sql.Timestamp;
004import java.time.LocalDateTime;
005
006import org.json.JSONArray;
007import org.json.JSONObject;
008
009
010public class ValueHelper {
011
012    public static Object getValue(AttributeType attributeType, Object source) {
013        Object value = null;
014        String stringValue = String.valueOf(source);
015        // conversion takes place here...
016        switch (attributeType) {
017            case BIG_INTEGER:
018                value = Long.valueOf(stringValue);
019                break;
020            case DATETIME:
021                if (!stringValue.endsWith("Z")) {
022                    stringValue = String.format("%s%s", stringValue, "Z");
023                }
024                value = stringValue;
025                break;
026            case DECIMAL:
027                value = Double.valueOf(stringValue);
028                break;
029            case INTEGER:
030                value = Integer.valueOf(stringValue);
031                break;
032            case VARCHAR:
033                value = stringValue;
034                break;
035            case TEXT:
036                value = stringValue;
037                break;
038            case JSON_BLOB:
039                if (stringValue.startsWith("{")) {
040                    value = new JSONObject(stringValue);
041                }
042                if (stringValue.startsWith("[")) {
043                    value = new JSONArray(stringValue);
044                }
045                break;
046        }
047        return value;
048    }
049    
050    public static Object toValue(AttributeType attributeType, Object source) {
051        Object value = null;
052        String stringValue = String.valueOf(source);
053        // conversion takes place here...
054        switch (attributeType) {
055            case BIG_INTEGER:
056                value = Long.valueOf(stringValue);
057                break;
058            case DATETIME:
059                if (stringValue.endsWith("Z")) {
060                    stringValue = stringValue.replace("Z", "");
061                }
062                value = Timestamp.valueOf(LocalDateTime.parse(stringValue));
063                break;
064            case DECIMAL:
065                value = Double.valueOf(stringValue);
066                break;
067            case INTEGER:
068                value = Integer.valueOf(stringValue);
069                break;
070            case VARCHAR:
071                value = stringValue;
072                break;
073            case TEXT:
074                value = stringValue;
075                break;
076            case JSON_BLOB:
077                value = stringValue;
078                break;
079        }
080        return value;
081    }
082    
083}
