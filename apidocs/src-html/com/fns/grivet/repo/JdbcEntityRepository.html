001package com.fns.grivet.repo;
002
003import java.sql.Timestamp;
004import java.sql.Types;
005import java.time.LocalDateTime;
006import java.util.ArrayList;
007import java.util.Arrays;
008import java.util.Collections;
009import java.util.Comparator;
010import java.util.List;
011
012import org.apache.commons.lang3.ObjectUtils;
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015import org.springframework.beans.factory.annotation.Autowired;
016import org.springframework.jdbc.core.JdbcTemplate;
017import org.springframework.jdbc.core.SqlParameterValue;
018import org.springframework.jdbc.core.SqlRowSetResultSetExtractor;
019import org.springframework.jdbc.core.simple.SimpleJdbcInsert;
020import org.springframework.jdbc.support.rowset.SqlRowSet;
021import org.springframework.stereotype.Repository;
022import org.springframework.util.Assert;
023
024import com.fns.grivet.model.Attribute;
025import com.fns.grivet.model.AttributeType;
026import com.fns.grivet.model.EntityAttributeValue;
027import com.fns.grivet.model.ValueHelper;
028import com.fns.grivet.query.DynamicQuery;
029import com.fns.grivet.query.QueryBuilder;
030import com.google.gwt.thirdparty.guava.common.collect.ImmutableMap;
031
032@Repository
033public class JdbcEntityRepository implements EntityRepository {
034
035    private final Logger log = LoggerFactory.getLogger(getClass());
036    
037    private final JdbcTemplate jdbcTemplate;
038    
039    @Autowired
040    public JdbcEntityRepository(JdbcTemplate jdbcTemplate) {
041        this.jdbcTemplate = jdbcTemplate;
042    }
043
044
045    @Override
046    public Long id(Integer cid) {
047        SimpleJdbcInsert insertEntity = new SimpleJdbcInsert(jdbcTemplate).withTableName("entity").usingGeneratedKeyColumns("eid").usingColumns("cid", "created_time");
048        return Long.valueOf(String.valueOf(insertEntity.executeAndReturnKey(ImmutableMap.of("cid", cid, "created_time", Timestamp.valueOf(LocalDateTime.now())))));
049    }
050
051    @Override
052    public void save(Long eid, Attribute attribute, AttributeType attributeType, Object rawValue) {
053        SimpleJdbcInsert insertEntityAttribtueValue = new SimpleJdbcInsert(jdbcTemplate).withTableName(String.join("_", "entityav", attributeType.getType())).usingColumns("eid", "aid", "val", "created_time");
054        Assert.isTrue(rawValue != null, String.format("Attempt to persist value failed! %s's value must not be null!", attribute.getName()));
055        Object value = ValueHelper.toValue(attributeType, rawValue);
056        insertEntityAttribtueValue.execute(ImmutableMap.of("eid", eid, "aid", attribute.getId(), "val", value, "created_time", Timestamp.valueOf(LocalDateTime.now())));
057    }
058
059    @Override
060    public List<EntityAttributeValue> find(Integer cid, LocalDateTime createdTimeStart,
061            LocalDateTime createdTimeEnd) {
062        String sql = QueryBuilder.newInstance().appendCreatedTimeRange().build();
063        log.trace(String.format("JdbcEntityRepository.find[sql=%s]", sql));
064        SqlRowSet rowSet = jdbcTemplate.query(sql, new SqlRowSetResultSetExtractor(), new SqlParameterValue(Types.INTEGER, cid), new SqlParameterValue(Types.TIMESTAMP, Timestamp.valueOf(createdTimeStart)), new SqlParameterValue(Types.TIMESTAMP, Timestamp.valueOf(createdTimeEnd)));
065        return mapRows(rowSet);
066    }
067
068    @Override
069    public List<EntityAttributeValue> executeDynamicQuery(Integer cid, DynamicQuery query) {
070        Assert.isTrue(query.areConjunctionsHomogenous(), "Query cannot be executed! All conjunctions must be homogenous!");
071        String sql = QueryBuilder.newInstance().append(query).build();
072        log.trace(String.format("JdbcEntityRepository.executeDynamicQuery[sql=%s]", sql));
073        List<SqlParameterValue> values = new ArrayList<>();
074        values.add(new SqlParameterValue(Types.INTEGER, cid));
075        values.addAll(Arrays.asList(query.asSqlParameterValues()));
076        SqlRowSet rowSet = jdbcTemplate.query(sql, new SqlRowSetResultSetExtractor(), values.toArray(new Object[values.size()]));
077        return mapRows(rowSet);
078    }
079        
080    private List<EntityAttributeValue> mapRows(SqlRowSet rowSet) {
081        List<EntityAttributeValue> result = new ArrayList<>();
082        EntityAttributeValue eav = null;
083        if (rowSet != null) {
084            while(rowSet.next()) {
085                eav = new EntityAttributeValue((Long) rowSet.getObject("eid"), (Integer) rowSet.getObject("attribute_id"), (String) rowSet.getObject("attribute_name"), rowSet.getObject("attribute_value"), ((Timestamp) rowSet.getObject("created_time")).toLocalDateTime());
086                result.add(eav);
087            }
088        }
089        Collections.sort(result, new EAVComparator());
090        return result;
091    }
092    
093    private static class EAVComparator implements Comparator<EntityAttributeValue> {
094
095        @Override
096        public int compare(EntityAttributeValue eav1, EntityAttributeValue eav2) {
097            return ObjectUtils.compare(eav1.getId(), eav2.getId());
098        }
099        
100    }
101
102}
