001package com.fns.grivet.config;
002
003import com.codahale.metrics.MetricRegistry;
004import com.fns.grivet.service.EntityService;
005import com.fns.grivet.service.PersistenceService;
006
007import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
008import org.springframework.cloud.stream.annotation.EnableBinding;
009import org.springframework.cloud.stream.messaging.Sink;
010
011@EnableBinding(Sink.class)
012@ConditionalOnProperty(prefix = "app.persistence.sink", name = "enabled", havingValue = "true")
013public class PersistenceSinkConfig {
014
015    public PersistenceService persistenceService(EntityService entityService, MetricRegistry metricRegistry) {
016        return new PersistenceService(entityService, metricRegistry);
017    }
018
019}
