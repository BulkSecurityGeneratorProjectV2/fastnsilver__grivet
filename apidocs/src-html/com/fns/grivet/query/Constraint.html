001package com.fns.grivet.query;
002
003import org.springframework.util.Assert;
004
005public class Constraint {
006
007    private final String attributeName;
008    private final Operator operator;
009    private final Conjunction conjunction;
010    private final String[] values;
011    
012    public Constraint(String attributeName, Operator operator, Conjunction conjunction, String[] values) {
013        this.attributeName = attributeName;
014        this.operator = operator;
015        this.conjunction = conjunction;
016        this.values = values;
017    }
018
019    // c=<attributeName>|<operator>|<value>|<conjunction>
020    public Constraint(String[] constraintParts) {
021        attributeName = constraintParts[0];
022        operator = Operator.fromValue(constraintParts[1]);
023        values = constraintParts[2].split("\\s*,\\s*");
024        if (operator.equals(Operator.BETWEEN)) {
025            Assert.isTrue(values.length == 2, "Operator [between] requires two values!");
026        } 
027        if (constraintParts.length == 4) {
028            if (constraintParts[3] == null) {
029                conjunction = null;
030            } else {
031                conjunction = Conjunction.fromValue(constraintParts[3]);
032            }
033        } else {
034            conjunction = null;
035        }
036    }
037
038    public String getAttributeName() {
039        return attributeName;
040    }
041    public Operator getOperator() {
042        return operator;
043    }
044
045    public Conjunction getConjunction() {
046        return conjunction;
047    }
048    
049    public String[] getValues() {
050        return values;
051    }
052    
053}
