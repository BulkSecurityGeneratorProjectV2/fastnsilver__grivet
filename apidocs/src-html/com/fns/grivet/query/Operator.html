001package com.fns.grivet.query;
002
003import java.util.Arrays;
004import java.util.List;
005import java.util.stream.Collectors;
006
007import org.springframework.util.Assert;
008
009
010public enum Operator {
011
012    STARTS_WITH("startsWith", "LIKE"),
013    ENDS_WITH("endsWith", "LIKE"),
014    CONTAINS("contains", "LIKE"),
015    EQUALS("equals", "="),
016    NOT_EQUAL_TO("notEqualTo", "<>"),
017    GREATER_THAN("greaterThan", ">"),
018    LESS_THAN("lessThan", "<"),
019    GREATER_THAN_OR_EQUAL_TO("greaterThanOrEqualTo", ">="),
020    LESS_THAN_OR_EQUAL_TO("lessThanOrEqualTo", "<="),
021    BETWEEN("between", "BETWEEN"),
022    IN("in", "IN");
023    
024    private String name;
025    private String symbol;
026    
027    Operator(String name, String symbol) {
028        this.name = name;
029        this.symbol = symbol;
030    }
031    
032    public String getName() {
033        return name;
034    }
035    
036    public String getSymbol() {
037        return symbol;
038    }
039    
040    public static Operator fromValue(String value) {
041        List<Operator> ops = Arrays.stream(Operator.values()).filter(o -> o.getName().equalsIgnoreCase(value)).collect(Collectors.toList());
042        Assert.notEmpty(ops, String.format("Invalid Operator [%s]", value));
043        return ops.get(0);
044    }
045    
046}
