16  package com.fns.grivet.security.oauth2;
17  
18  
19  import java.util.Collection;
20  
21  import org.springframework.security.core.Authentication;
22  import org.springframework.security.oauth2.provider.AuthorizationRequest;
23  import org.springframework.security.oauth2.provider.ClientDetails;
24  import org.springframework.security.oauth2.provider.ClientDetailsService;
25  import org.springframework.security.oauth2.provider.ClientRegistrationException;
26  import org.springframework.security.oauth2.provider.approval.ApprovalStoreUserApprovalHandler;
27  
28  public class GrivetUserApprovalHandler extends ApprovalStoreUserApprovalHandler {
29  
30      private boolean useApprovalStore = true;
31  
32      private ClientDetailsService clientDetailsService;
33  
34      /**
35       * Service to load client details (optional) for auto approval checks.
36       * 
37       * @param clientDetailsService a client details service
38       */
39      @Override
40      public void setClientDetailsService(ClientDetailsService clientDetailsService) {
41          this.clientDetailsService = clientDetailsService;
42          super.setClientDetailsService(clientDetailsService);
43      }
44  
45      /**
46       * @param useApprovalStore the useTokenServices to set
47       */
48      public void setUseApprovalStore(boolean useApprovalStore) {
49          this.useApprovalStore = useApprovalStore;
50      }
51  
52      /**
53       * Allows automatic approval for a white list of clients in the implicit grant case.
54       * 
55       * @param authorizationRequest The authorization request.
56       * @param userAuthentication the current user authentication
57       * 
58       * @return An updated request if it has already been approved by the current user.
59       */
60      @Override
61      public AuthorizationRequest checkForPreApproval(AuthorizationRequest authorizationRequest,
62              Authentication userAuthentication) {
63  
64          boolean approved = false;
65          // If we are allowed to check existing approvals this will short circuit the decision
66          if (useApprovalStore) {
67              authorizationRequest = super.checkForPreApproval(authorizationRequest, userAuthentication);
68              approved = authorizationRequest.isApproved();
69          }
70          else {
71              if (clientDetailsService != null) {
72                  Collection<String> requestedScopes = authorizationRequest.getScope();
73                  try {
74                      ClientDetails client = clientDetailsService
75                              .loadClientByClientId(authorizationRequest.getClientId());
76                      for (String scope : requestedScopes) {
77                          if (client.isAutoApprove(scope)) {
78                              approved = true;
79                              break;
80                          }
81                      }
82                  }
83                  catch (ClientRegistrationException e) {
84                  }
85              }
86          }
87          authorizationRequest.setApproved(approved);
88  
89          return authorizationRequest;
90  
91      }
92  
93  }
