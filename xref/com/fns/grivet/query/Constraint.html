3   import org.springframework.util.Assert;
4   
5   public class Constraint {
6   
7       private final String attributeName;
8       private final Operator operator;
9       private final Conjunction conjunction;
10      private final String[] values;
11      
12      public Constraint(String attributeName, Operator operator, Conjunction conjunction, String[] values) {
13          this.attributeName = attributeName;
14          this.operator = operator;
15          this.conjunction = conjunction;
16          this.values = values;
17      }
18  
19      // c=<attributeName>|<operator>|<value>|<conjunction>
20      public Constraint(String[] constraintParts) {
21          attributeName = constraintParts[0];
22          operator = Operator.fromValue(constraintParts[1]);
23          values = constraintParts[2].split("\\s*,\\s*");
24          if (operator.equals(Operator.BETWEEN)) {
25              Assert.isTrue(values.length == 2, "Operator [between] requires two values!");
26          } 
27          if (constraintParts.length == 4) {
28              if (constraintParts[3] == null) {
29                  conjunction = null;
30              } else {
31                  conjunction = Conjunction.fromValue(constraintParts[3]);
32              }
33          } else {
34              conjunction = null;
35          }
36      }
37  
38      public String getAttributeName() {
39          return attributeName;
40      }
41      public Operator getOperator() {
42          return operator;
43      }
44  
45      public Conjunction getConjunction() {
46          return conjunction;
47      }
48      
49      public String[] getValues() {
50          return values;
51      }
52      
53  }
