1   package com.fns.grivet.query;
2   
3   import java.util.Arrays;
4   import java.util.List;
5   import java.util.stream.Collectors;
6   
7   import org.springframework.util.Assert;
8   
9   
10  public enum Operator {
11  
12      STARTS_WITH("startsWith", "LIKE"),
13      ENDS_WITH("endsWith", "LIKE"),
14      CONTAINS("contains", "LIKE"),
15      EQUALS("equals", "="),
16      NOT_EQUAL_TO("notEqualTo", "<>"),
17      GREATER_THAN("greaterThan", ">"),
18      LESS_THAN("lessThan", "<"),
19      GREATER_THAN_OR_EQUAL_TO("greaterThanOrEqualTo", ">="),
20      LESS_THAN_OR_EQUAL_TO("lessThanOrEqualTo", "<="),
21      BETWEEN("between", "BETWEEN"),
22      IN("in", "IN");
23      
24      private String name;
25      private String symbol;
26      
27      Operator(String name, String symbol) {
28          this.name = name;
29          this.symbol = symbol;
30      }
31      
32      public String getName() {
33          return name;
34      }
35      
36      public String getSymbol() {
37          return symbol;
38      }
39      
40      public static Operator fromValue(String value) {
41          List<Operator> ops = Arrays.stream(Operator.values()).filter(o -> o.getName().equalsIgnoreCase(value)).collect(Collectors.toList());
42          Assert.notEmpty(ops, String.format("Invalid Operator [%s]", value));
43          return ops.get(0);
44      }
45      
46  }
