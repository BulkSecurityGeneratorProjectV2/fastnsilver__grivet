11  import org.springframework.web.bind.annotation.RequestBody;
12  import org.springframework.web.bind.annotation.RequestMapping;
13  import org.springframework.web.bind.annotation.RequestMethod;

15  import org.springframework.web.bind.annotation.RestController;
16  import org.springframework.web.util.UriComponentsBuilder;
17  
18  import com.fns.grivet.query.NamedQuery;
19  import com.fns.grivet.query.QueryType;
20  import com.fns.grivet.service.NamedQueryService;
21  
22  
23  @RestController
24  @RequestMapping("/query")
25  public class NamedQueryController {
26  
27      private final NamedQueryService namedQueryService;
28      
29      @Autowired
30      public NamedQueryController(NamedQueryService namedQueryService) {
31          this.namedQueryService = namedQueryService;
32      }
33      
34      @RequestMapping(method=RequestMethod.POST, 
35              consumes=MediaType.APPLICATION_JSON_VALUE, produces=MediaType.APPLICATION_JSON_VALUE)
36      public ResponseEntity<?> create(@RequestBody NamedQuery query) {
37          ResponseEntity<?> result = ResponseEntity.unprocessableEntity().build();
38          Assert.isTrue(StringUtils.isNotBlank(query.getName()), "Query name must not be null, empty or blank.");
39          Assert.notNull(query.getQuery(), "Query string must not be null!");
40          Assert.isTrue(isSupportedQuery(query), "Query must start with either CALL or SELECT");
41          // check for named parameters; but only if they exist
42          if (!query.getParams().isEmpty()) {
43              query.getParams().keySet()
44                  .forEach(k -> Assert.isTrue(query.getQuery().contains(String.format(":%s", k)), String.format("Query must contain named parameter [%s]", k)));
45          }
46          namedQueryService.create(query);
47          UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
48          if (query.getParams().isEmpty()) {
49              result = ResponseEntity.created(ucb.path("/query/{name}").buildAndExpand(query.getName()).toUri()).build();
50          } else {
51              result = ResponseEntity.noContent().build(); 
52          }
53          return result;
54      }
55      
56      @RequestMapping(value="/{name}", produces=MediaType.APPLICATION_JSON_VALUE)
57      public ResponseEntity<?> get(@PathVariable("name") String name, @RequestParam MultiValueMap<String, ?> parameters) {
58          return ResponseEntity.ok(namedQueryService.get(name, parameters));
59      }
60      
61      @RequestMapping(value="", produces=MediaType.APPLICATION_JSON_VALUE)
62      public ResponseEntity<?> all(@RequestParam("showAll") String showAll) {
63          JSONArray payload = namedQueryService.all();
64          return ResponseEntity.ok(payload.toString());
65      }
66      
67      private boolean isSupportedQuery(NamedQuery query) {
68          boolean result = false;
69          if ((query.getQuery().toUpperCase().startsWith("SELECT") && query.getType().equals(QueryType.SELECT)) 
70                  || (query.getQuery().toUpperCase().startsWith("CALL") && query.getType().equals(QueryType.SPROC))) {
71              result = true;
72          }
73          return result;
74      }
75  }
