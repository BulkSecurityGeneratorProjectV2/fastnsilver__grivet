26  @RestController
27  @RequestMapping("/register")
28  public class ClassRegistryController {
29  
30      private final ClassRegistryService classRegistryService;
31      
32      @Autowired
33      public ClassRegistryController(ClassRegistryService classRegistryService) {
34          this.classRegistryService = classRegistryService;
35      }
36      
37      @RequestMapping(method=RequestMethod.POST, 
38              consumes=MediaType.APPLICATION_JSON_VALUE, produces=MediaType.APPLICATION_JSON_VALUE)
39      public ResponseEntity<?> register(HttpServletRequest request) throws IOException {
40          String json = IOUtils.toString(request.getInputStream(), "UTF-8");
41          Assert.isTrue(json.startsWith("{") || json.startsWith("["), "Registration requests must be valid JSON starting with either a { or [!");
42          ResponseEntity<?> result = ResponseEntity.unprocessableEntity().build();
43          if (json.startsWith("{")) {
44              if (request.getParameterMap().containsKey("linkSchema") 
45                      && classRegistryService.isJsonSchema(new JSONObject(json))) {
46                  result = linkSchema(json);
47              } else {
48                  result = registerSingleType(json);
49              }
50          }
51          if (json.startsWith("[")) {
52              result = registerMultipleTypes(json);
53          }
54          return result;
55      }
56      
57      private ResponseEntity<?> linkSchema(String json) {
58          String id = classRegistryService.linkSchema(new JSONObject(json)).getName();
59          return ResponseEntity.ok(String.format("JSON Schema for type [%s] linked!  Store requests for this type will be validated henceforth!", id));
60      }
61      
62      private ResponseEntity<?> unlinkSchema(String type) {
63          classRegistryService.unlinkSchema(type);
64          return ResponseEntity.ok(String.format("JSON Schema for type [%s] unlinked!  Store requests for this type will no longer be validated!", type));
65      }
66      
67      private ResponseEntity<?> registerSingleType(String json) {
68          String type = classRegistryService.register(new JSONObject(json));
69          UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
70          return ResponseEntity.created(ucb.path("/register/{type}").buildAndExpand(type).toUri()).build();
71      }
72      
73      private ResponseEntity<?> registerMultipleTypes(String json) {
74          JSONArray jsonArray = new JSONArray(json);
75          JSONObject jsonObject = null;
76          String type = null;
77          HttpHeaders headers = new HttpHeaders();
78          UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
79          int numberOfTypesToRegister = jsonArray.length();
80          URI location = null;
81          int errorCount = 0;
82          // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
83          for (int i = 0; i < numberOfTypesToRegister; i++) {
84              jsonObject = jsonArray.getJSONObject(i);
85              try {
86                  type = classRegistryService.register(jsonObject);
87                  location = ucb.path("/register/{type}").buildAndExpand(type).toUri();
88                  if (numberOfTypesToRegister == 1) {
89                      headers.setLocation(location); 
90                  } else {
91                      headers.set(String.format("Location[%s]",String.valueOf(i+1)), location.toASCIIString());
92                  }
93              } catch (Exception e) {
94                  if (numberOfTypesToRegister == 1) {
95                      throw e;
96                  }
97                  headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
98                  errorCount++;
99              }
100         }
101         HttpStatus status = (errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED;
102         return new ResponseEntity<>(headers, status);
103     }
104     
105     @RequestMapping(value="/{type}", produces=MediaType.APPLICATION_JSON_VALUE)
106     public ResponseEntity<?> get(@PathVariable("type") String type) {
107         JSONObject payload = classRegistryService.get(type);
108         return ResponseEntity.ok(payload.toString());
109     }
110     
111     @RequestMapping(value="/{type}", method=RequestMethod.PUT, produces=MediaType.APPLICATION_JSON_VALUE)
112     public ResponseEntity<?> unlinkSchema(@PathVariable("type") String type, HttpServletRequest request) {
113         Assert.isTrue(request.getParameterMap().containsKey("unlinkSchema"), "Operation not supported!"); 
114         return unlinkSchema(type);
115     }
116     
117     @RequestMapping(value="/", produces=MediaType.APPLICATION_JSON_VALUE)
118     public ResponseEntity<?> all() {
119         JSONArray payload = classRegistryService.all();
120         return ResponseEntity.ok(payload.toString());
121     }
