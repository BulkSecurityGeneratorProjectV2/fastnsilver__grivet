20  import org.json.JSONObject;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.http.MediaType;
25  import org.springframework.http.ResponseEntity;
26  import org.springframework.security.access.prepost.PreAuthorize;
27  import org.springframework.web.bind.annotation.PathVariable;
28  import org.springframework.web.bind.annotation.RequestBody;
29  import org.springframework.web.bind.annotation.RequestMapping;
30  import org.springframework.web.bind.annotation.RequestMethod;
31  import org.springframework.web.bind.annotation.RestController;
32  
33  import java.io.IOException;
34  
35  import io.swagger.annotations.Api;
36  import io.swagger.annotations.ApiOperation;
37  import io.swagger.annotations.ApiParam;
38  import io.swagger.annotations.ApiResponse;
39  import io.swagger.annotations.ApiResponses;
40  
41  
42  /**
43   * Provides end-points for linking and unlinking JSON Schema to pre-registered types
44   * 
45   * @author Chris Phillipson
46   */
47  @RestController
48  @RequestMapping("/type/schema")
49  @Api(value = "type/schema", produces = "application/json")
50  public class SchemaController {
51  
52      private final Logger log = LoggerFactory.getLogger(getClass());
53          
54      private final SchemaService schemaService;
55      
56      @Autowired
57      public SchemaController(SchemaService schemaService) {
58          this.schemaService = schemaService;
59      }
60      
61      
62      @PreAuthorize(value = "hasRole(@roles.ADMIN)")
63      @RequestMapping(value = "/link", method = RequestMethod.POST, 
64              consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
65      @ApiOperation(httpMethod = "POST", notes = "Link a JSON Schema to a pre-registered type.", value = "/type/schema/link")
66      @ApiResponses(value = { 
67              @ApiResponse(code = 200, message = "Successfully link JSON Schema to registered type."),
68              @ApiResponse(code = 400, message = "Bad request."),
69              @ApiResponse(code = 422, message = "Unprocessable entity (e.g., invalid JSON schema)."),
70              @ApiResponse(code = 500, message = "Internal server error.")
71              })
72      public ResponseEntity<?> linkSchema(@RequestBody String payload) throws IOException {
73          JSONObject json = new JSONObject(payload);
74          if (schemaService.isJsonSchema(json)) {
75              String id = schemaService.linkSchema(json).getName();
76              String message = String.format("JSON Schema for type [%s] linked!  Store requests for this type will be validated henceforth!", id);
77              log.info(message);
78              return ResponseEntity.ok(message);
79          } 
80          return ResponseEntity.unprocessableEntity().build();
81      }
82      
83      @PreAuthorize(value = "hasRole(@roles.ADMIN)")
84      @RequestMapping(value = "/unlink/{type}", method = RequestMethod.PUT, produces = MediaType.APPLICATION_JSON_VALUE)
85      @ApiOperation(httpMethod = "PUT", notes = "Unlink JSON Schema from a pre-registered type.", value = "/type/schema/unlink/{type}")
86      @ApiResponses(value = { 
87              @ApiResponse(code = 200, message = "Unlinked JSON Schema from type."),
88              @ApiResponse(code = 400, message = "Bad request."),
89              @ApiResponse(code = 500, message = "Internal server error.")
90              })
91      public ResponseEntity<?> unlinkSchema(
92              @ApiParam(value = "Type name", required = true)
93              @PathVariable("type") String type) {
94          schemaService.unlinkSchema(type);
95          String message = String.format("JSON Schema for type [%s] unlinked!  Store requests for this type will no longer be validated!", type);
96          log.info(message);
97          return ResponseEntity.ok(message);
98      }
99              
100 }
