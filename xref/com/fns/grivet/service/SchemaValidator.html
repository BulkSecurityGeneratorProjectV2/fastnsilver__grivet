10  import com.fasterxml.jackson.databind.JsonNode;
11  import com.fasterxml.jackson.databind.ObjectMapper;
12  import com.fns.grivet.repo.ClassRepository;
13  import com.github.fge.jsonschema.core.exceptions.ProcessingException;
14  import com.github.fge.jsonschema.core.report.ProcessingReport;
15  import com.github.fge.jsonschema.main.JsonSchema;
16  import com.github.fge.jsonschema.main.JsonSchemaFactory;
17  
18  @Component
19  class SchemaValidator {
20      
21      private ClassRepository classRepository;
22      private ObjectMapper objectMapper;
23      
24      @Autowired
25      public SchemaValidator(ClassRepository classRepository, ObjectMapper objectMapper) {
26          this.classRepository = classRepository;
27          this.objectMapper = objectMapper;
28      }
29  
30      public ProcessingReport validate(String type, JSONObject payload) throws SchemaValidationException {
31          try {
32              JsonSchemaFactory factory = JsonSchemaFactory.byDefault();
33              com.fns.grivet.model.Class c = classRepository.findByName(type);
34              Assert.notNull(type, String.format("Type [%s] does exist! Schema cannot be retrieved!", type));
35              JsonNode schemaAsJsonNode = objectMapper.readTree(c.getJsonSchema());
36              final JsonSchema schema = factory.getJsonSchema(schemaAsJsonNode);
37              JsonNode instance = objectMapper.readTree(payload.toString());
38              Assert.notNull(schema, String.format("Schema not found for type [%s]!", type));
39              Assert.notNull(instance, String.format("Problem generating JsonNode for payload\n\n%s!", payload.toString()));
40              return schema.validate(instance);
41          } catch (ProcessingException | IOException e) {
42              throw new SchemaValidationException(String.format("Problem validating [%s] against JSON Schema", type), e);
43          }
44      }
45      
46  }
