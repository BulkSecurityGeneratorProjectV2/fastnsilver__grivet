28      private static final String ID = "id";
29      private static final String SCHEMA = "$schema";
30      private static final String TYPE = "type";
31      private static final String DESCRIPTION = "description";
32      private static final String ATTRIBUTES = "attributes";
33      private static final String PROPERTIES = "properties";

35      private final AttributeRepository attributeRepository;

37      private final ClassRepository classRepository;
38      private final ClassAttributeRepository classAttributeRepository;

40      @Autowired
41      public ClassRegistryService(AttributeRepository attributeRepository,
42              AttributeTypeRepository attributeTypeRepository, ClassRepository classRepository,
43              ClassAttributeRepository classAttributeRepository) {
44          this.attributeRepository = attributeRepository;
45          this.attributeTypeRepository = attributeTypeRepository;
46          this.classRepository = classRepository;
47          this.classAttributeRepository = classAttributeRepository;
48      }

50      @Transactional
51      // USE WITH CAUTION! Consider that w/ FK, type and all related data will be destroyed
52      public void deregister(String type) {
53          com.fns.grivet.model.Class c = classRepository.findByName(type);
54          if (c != null) {
55              classRepository.delete(c);
56          }
57      }
58      
59      @Transactional
60      public String register(JSONObject payload) {
61          String type = payload.getString(TYPE);
62          String description = payload.optString(DESCRIPTION);
63          com.fns.grivet.model.Class c = classRepository.findByName(type);
64          if (c == null) {
65              c = classRepository.save(new com.fns.grivet.model.Class(type, description));
66              JSONObject attributes = payload.getJSONObject(ATTRIBUTES);
67              Set<String> attributeNames = attributes.keySet();
68              Assert.notEmpty(attributeNames, String.format("[%s] must declare at least one attribute in registration request!", type));
69              Attribute a = null;
70              String at = null;
71              AttributeType attributeType = null;
72              // TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
73              for (String attributeName: attributeNames) {
74                  a = attributeRepository.findByName(attributeName);
75                  if (a == null) {
76                      a = attributeRepository.save(new Attribute(attributeName));
77                  }
78                  at = attributes.getString(attributeName);
79                  Assert.notNull(at, String.format("[%s].[%s] must declare an attribute type in registration request!", type, attributeName));
80                  attributeType = attributeTypeRepository.findByType(at);
81                  Assert.notNull(attributeType, String.format("Attribute type [%s] is not supported!", at));
82                  classAttributeRepository.save(new ClassAttribute(c.getId(), a.getId(), attributeType.getId()));
83              }
84          } else {
85              throw new DuplicateKeyException(String.format("Type [%s] already registered!", type));
86          }
87          return type;
88      }
89  
90      @Transactional(readOnly=true)
91      public JSONObject get(String type) {
92          com.fns.grivet.model.Class c = classRepository.findByName(type);
93          Assert.notNull(c, String.format("Type [%s] is not registered!", type));
94          JSONObject result = new JSONObject();
95          result.put("type", type);
96          String description = c.getDescription();
97          result.put("description", description != null ? description: "");
98          List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
99          Assert.notEmpty(cas, String.format("Type [%s] does not have any attributes registered!", type));
100         JSONObject attributes = new JSONObject();
101         result.put("attributes", attributes);
102         Attribute a = null;
103         AttributeType at = null;
104         for (ClassAttribute ca: cas) {
105             a = attributeRepository.findOne(ca.getAid());
106             Assert.notNull(a, String.format("Attribute id [%s] is not registered!", ca.getAid()));
107             at = attributeTypeRepository.findOne(ca.getTid());
108             Assert.notNull(a, String.format("Attribute Type id [%s] is not registered!", ca.getTid()));
109             attributes.put(a.getName(), at.getType());
110         }
111         return result;
112     }
113     
114     @Transactional(readOnly=true)
115     public JSONArray all() {
116         JSONArray result = new JSONArray();
117         Iterable<com.fns.grivet.model.Class> iterable = classRepository.findAll();
118         Assert.notNull(iterable, "No types are registered!");
119         Iterator<com.fns.grivet.model.Class> it = iterable.iterator();
120         com.fns.grivet.model.Class c = null;
121         while(it.hasNext()) {
122            c = it.next();
123            result.put(get(c.getName()));
124         }
125         return result;
126     }

128     public boolean isJsonSchema(JSONObject payload) {
129         boolean result = false;
130         String schema = payload.optString(SCHEMA);
131         String type = payload.optString(TYPE);
132         String id = payload.optString(ID);
133         String props = payload.optString(PROPERTIES);
134         if (schema.equals("http://json-schema.org/draft-04/schema#") 
135                 && type.equals("object") 
136                 && StringUtils.isNotBlank(id) 
137                 && StringUtils.isNotBlank(props)) {
138             result = true;
139         }
140         return result;
141     }

143     @Transactional
144     public com.fns.grivet.model.Class linkSchema(JSONObject payload) {
145         String type = payload.getString(ID);
146         com.fns.grivet.model.Class c = classRepository.findByName(type);
147         Assert.notNull(c, String.format("Type [%s] must be registered before linking a JSON Schema!", type));
148         c.setValidatable(true);
149         c.setJsonSchema(payload.toString());
150         c.setUpdatedTime(LocalDateTime.now());
151         classRepository.save(c);
152         return c;
153     }
154     
155     @Transactional
156     public com.fns.grivet.model.Class unlinkSchema(String type) {
157         com.fns.grivet.model.Class c = classRepository.findByName(type);
158         Assert.notNull(c, String.format("Type [%s] must be registered before unlinking a JSON Schema!", type));
159         c.setValidatable(false);
160         c.setJsonSchema(null);
161         c.setUpdatedTime(LocalDateTime.now());
162         classRepository.save(c);
163         return c;
164     }
