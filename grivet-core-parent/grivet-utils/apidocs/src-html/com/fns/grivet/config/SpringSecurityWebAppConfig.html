001package com.fns.grivet.config;
002
003import static com.stormpath.spring.config.StormpathWebSecurityConfigurer.stormpath;
004
005import com.stormpath.spring.config.EnableStormpathWebSecurity;
006
007import org.springframework.context.annotation.Configuration;
008import org.springframework.context.annotation.Profile;
009import org.springframework.security.config.annotation.web.builders.HttpSecurity;
010import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
011
012@Profile("secure")
013@Configuration
014@EnableStormpathWebSecurity
015// https://docs.stormpath.com/java/spring-boot-web/tutorial.html#spring-security-meet-stormpath 
016// says we don't need to add the second annotation above, but if we don't add it we get a 
017// org.springframework.beans.factory.BeanCreationException: Could not autowire field: 
018// com.stormpath.spring.oauth.Oauth2AuthenticationSpringSecurityProcessingFilter
019public class SpringSecurityWebAppConfig extends WebSecurityConfigurerAdapter {
020
021    @Override
022    protected void configure(HttpSecurity http) throws Exception {
023        http
024            .apply(stormpath()).and()
025            .authorizeRequests()
026            .antMatchers("/").permitAll();
027    }
028}
