001package com.fns.grivet.config;
002
003import org.springframework.beans.factory.annotation.Autowired;
004import org.springframework.context.annotation.Profile;
005import org.springframework.core.env.Environment;
006import org.springframework.stereotype.Component;
007
008@Profile("secure")
009@Component
010public class Roles {
011
012    public final String USER;
013    public final String ADMIN;
014
015    @Autowired
016    public Roles(Environment env) {
017        USER = env.getProperty("stormpath.authorized.group.user");
018        ADMIN = env.getProperty("stormpath.authorized.group.admin");
019    }
020}
