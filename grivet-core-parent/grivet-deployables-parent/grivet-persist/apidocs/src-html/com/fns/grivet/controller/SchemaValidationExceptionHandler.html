001package com.fns.grivet.controller;
002
003import javax.servlet.http.HttpServletRequest;
004
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007import org.springframework.http.ResponseEntity;
008import org.springframework.web.bind.annotation.ControllerAdvice;
009import org.springframework.web.bind.annotation.ExceptionHandler;
010
011import com.fns.grivet.model.ErrorResponse;
012import com.fns.grivet.service.SchemaValidationException;
013
014@ControllerAdvice
015public class SchemaValidationExceptionHandler {
016
017    private final Logger log = LoggerFactory.getLogger(getClass());
018
019    @ExceptionHandler({ SchemaValidationException.class })
020    protected ResponseEntity<ErrorResponse> invalid(SchemaValidationException e, HttpServletRequest hsr) {
021        ErrorResponse er = new ErrorResponse(hsr.getMethod(), hsr.getRequestURI(), hsr.getQueryString(),
022                e.getProcessingFailures());
023        log.error(er.toString());
024        return ResponseEntity.badRequest().body(er);
025    }
026}
