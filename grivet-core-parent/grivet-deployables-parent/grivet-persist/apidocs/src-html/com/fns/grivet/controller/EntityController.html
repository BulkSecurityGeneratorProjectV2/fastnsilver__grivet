055/**
056 * Provides end-points for type storage and retrieval
057 * 
058 * @author Chris Phillipson
059 */
060@RestController
061@RequestMapping("/store")
062@Api(value = "store", produces = "application/json")
063public class EntityController {
064
065    private final Logger log = LoggerFactory.getLogger(getClass());
066    
067    @Value("${grivet.store.batch-size:100}")
068    int batchSize;
069    
070    private final EntityService entityService;
071    
072    private final MetricRegistry metricRegistry;
073
074    @Autowired
075    public EntityController(EntityService entityService, MetricRegistry metricRegistry) {
076        this.entityService = entityService;
077        this.metricRegistry = metricRegistry;
078    }
079    
080    @Profile("!no-http")
081    @PreAuthorize("hasRole(@roles.ADMIN) or hasRole(@roles.USER)")
082    @RequestMapping(value = "/{type}", method = RequestMethod.POST, 
083            consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
084    @ApiOperation(httpMethod = "POST", notes = "Store a type.", value = "/store/{type}")
085    @ApiResponses({ @ApiResponse(code = 204, message = "Successfully store type."),
086            @ApiResponse(code = 202, message = "Partial success. Error details for type(s) that could not be registered."),
087            @ApiResponse(code = 400, message = "Bad request."),
088            @ApiResponse(code = 500, message = "Internal server error.") })
089    public ResponseEntity<?> createSingle(@PathVariable("type") String type, @RequestBody JSONObject json)
090            throws IOException {
091        entityService.create(type, json);
092        metricRegistry.counter(MetricRegistry.name("store", type, "count")).inc();
093        log.info("Successfully stored type [{}]", type);
094        return ResponseEntity.noContent().build();
095    }
096
097    @Profile("!no-http")
098    @PreAuthorize("hasRole(@roles.ADMIN) or hasRole(@roles.USER)")
099    @RequestMapping(value = "/batch/{type}", method = RequestMethod.POST, consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
100    @ApiOperation(httpMethod = "POST", notes = "Store multiple types.", value = "/store/batch/{type}")
101    @ApiResponses({ @ApiResponse(code = 204, message = "Successfully store types."),
102            @ApiResponse(code = 202, message = "Partial success. Error details for type(s) that could not be registered."),
103            @ApiResponse(code = 400, message = "Bad request."),
104            @ApiResponse(code = 500, message = "Internal server error.") })
105    public ResponseEntity<?> createMultiple(@PathVariable("type") String type, @RequestBody JSONArray json)
106            throws IOException, JSONException {
107        int numberOfTypesToCreate = json.length();
108        Assert.isTrue(numberOfTypesToCreate <= batchSize,
109                String.format(
110                        "The total number of entries in a request must not exceed %d! Your store request contained [%d] entries.",
111                        batchSize, numberOfTypesToCreate));
112        int errorCount = 0;
113        JSONObject jsonObject = null;
114        HttpHeaders headers = new HttpHeaders();
115        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
116        for (int i = 0; i < numberOfTypesToCreate; i++) {
117            try {
118                jsonObject = json.getJSONObject(i);
119                entityService.create(type, jsonObject);
120                metricRegistry.counter(MetricRegistry.name("store", type, "count")).inc();
121                log.info("Successfully stored type [{}]", type);
122            } catch (Exception e) {
123                String message = LogUtil.toLog(jsonObject, String.format("Problems storing type! Portion of payload @ index[%d]\n", i+1));
124                log.error(message, e);
125                if (numberOfTypesToCreate == 1) {
126                    throw e;
127                }
128                headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
129                errorCount++;
130            }
131        }
132        return new ResponseEntity<>(headers, ((errorCount == 0) ? HttpStatus.NO_CONTENT : HttpStatus.ACCEPTED));
133    }
134    
135    @PreAuthorize("hasRole(@roles.ADMIN) or hasRole(@roles.USER)")
136    @RequestMapping(value = "/{type}", produces = MediaType.APPLICATION_JSON_VALUE)
137    @ApiOperation(httpMethod = "GET", notes = "Retrieve type matching criteria.", value = "/store/{type}")
138    @ApiResponses({ @ApiResponse(code = 200, message = "Successfully retrieve type matching criteria."),
139            @ApiResponse(code = 400, message = "Bad request."),
140            @ApiResponse(code = 500, message = "Internal server error.") })
141    public ResponseEntity<?> get(@PathVariable("type") String type,
142            @RequestParam(value = "createdTimeStart", required = false) LocalDateTime createdTimeStart,
143            @RequestParam(value = "createdTimeEnd", required = false) LocalDateTime createdTimeEnd,
144            HttpServletRequest request) throws JsonProcessingException {
145        LocalDateTime start = createdTimeStart == null ? LocalDateTime.now().minusDays(7) : createdTimeStart;
146        LocalDateTime end = createdTimeEnd == null ? LocalDateTime.now() : createdTimeEnd;
147        Assert.isTrue(ChronoUnit.SECONDS.between(start, end) >= 0, "Store request constraint createdTimeStart must be earlier or equal to createdTimeEnd!");
148        return ResponseEntity.ok(entityService.findByCreatedTime(type, start, end, request.getParameterMap()));
149    }
150    
151}
