055/**
056 * Provides end-points for type storage and retrieval
057 *
058 * @author Chris Phillipson
059 */
060@RestController
061@RequestMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
062public class EntityController {
063
064    private final Logger log = LoggerFactory.getLogger(getClass());
065
066    @Value("${grivet.store.batch-size:100}")
067    private int batchSize;
068
069    private final EntityService entityService;
070
071    private final MetricRegistry metricRegistry;
072
073    @Autowired
074    public EntityController(EntityService entityService, MetricRegistry metricRegistry) {
075        this.entityService = entityService;
076        this.metricRegistry = metricRegistry;
077    }
078
079    @Profile("!pipeline")
080    @PreAuthorize("hasAuthority('write:type')")
081    @PostMapping("/api/v1/type")
082    public ResponseEntity<?> createOne(@RequestHeader("Type") String type, @RequestBody JSONObject json) {
083        Long oid = entityService.create(type, json);
084        URI location = UriComponentsBuilder.newInstance().path("/api/v1/type").queryParam("oid", oid).build().toUri();
085        metricRegistry.counter(MetricRegistry.name("store", "create", type, "count")).inc();
086        log.info("Successfully created type [{}]", type);
087        return ResponseEntity.created(location).build();
088    }
089
090    @Profile("!pipeline")
091    @PreAuthorize("hasAuthority('write:type')")
092    @PostMapping("/api/v1/types")
093    public ResponseEntity<?> createMultiple(@RequestHeader("Type") String type, @RequestBody JSONArray json) {
094        int numberOfTypesToCreate = json.length();
095        Assert.isTrue(numberOfTypesToCreate <= batchSize,
096                String.format(
097                        "The total number of entries in a request must not exceed %d! Your store request contained [%d] entries.",
098                        batchSize, numberOfTypesToCreate));
099        int errorCount = 0;
100        JSONObject jsonObject = null;
101        HttpHeaders headers = new HttpHeaders();
102        URI location = null;
103        Long oid = null;
104        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
105        for (int i = 0; i < numberOfTypesToCreate; i++) {
106            try {
107                jsonObject = json.getJSONObject(i);
108                oid = entityService.create(type, jsonObject);
109                location = UriComponentsBuilder.newInstance().path("/api/v1/type").queryParam("oid", oid).build().toUri();
110                if (numberOfTypesToCreate == 1) {
111                    headers.setLocation(location);
112                } else {
113                    headers.set(String.format("Location[%s]", String.valueOf(i + 1)), location.toASCIIString());
114                }
115                metricRegistry.counter(MetricRegistry.name("store", "create", type, "count")).inc();
116                log.info("Successfully created type [{}]", type);
117            } catch (Exception e) {
118                String message = LogUtil.toLog(jsonObject, String.format("Problems storing type! Portion of payload @ index[%d]\n", i+1));
119                log.error(message, e);
120                if (numberOfTypesToCreate == 1) {
121                    throw e;
122                }
123                headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
124                errorCount++;
125            }
126        }
127        return new ResponseEntity<>(headers, ((errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED));
128    }
129
130    @Profile("!pipeline")
131    @PreAuthorize("hasAuthority('write:type')")
132    @PatchMapping("/api/v1/type")
133    public ResponseEntity<?> updateOne(
134            @RequestParam(value = "oid", required = true) Long oid,
135            @RequestBody JSONObject json) {
136        String type = entityService.update(oid, json);
137        HttpHeaders headers = new HttpHeaders();
138        URI location = UriComponentsBuilder.newInstance().path("/api/v1/type").queryParam("oid", oid).build().toUri();
139        headers.setLocation(location);
140        metricRegistry.counter(MetricRegistry.name("store", "update", type, "count")).inc();
141        log.info("Successfully updated type [{}]", type);
142        return ResponseEntity.ok().headers(headers).build();
143    }
144
145    @Profile("!pipeline")
146    @PreAuthorize("hasAuthority('delete:type')")
147    @DeleteMapping("/api/v1/type")
148    public ResponseEntity<?> deleteOne(
149            @RequestParam(value = "oid", required = true) Long oid) {
150        String type = entityService.delete(oid);
151        metricRegistry.counter(MetricRegistry.name("store", "delete", type, "count")).inc();
152        log.info("Successfully delete type [{}]", type);
153        return ResponseEntity.ok().build();
154    }
155
156    @PreAuthorize("hasAuthority('read:type')")
157    @GetMapping("/api/v1/type/{type}")
158    public ResponseEntity<?> fetch(@PathVariable("type") String type,
159            @RequestParam(value = "createdTimeStart", required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE_TIME) LocalDateTime createdTimeStart,
160            @RequestParam(value = "createdTimeEnd", required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE_TIME) LocalDateTime createdTimeEnd,
161            @RequestParam(value = "noAudit", defaultValue = "false") boolean noAudit,
162            HttpServletRequest request) throws JsonProcessingException {
163        LocalDateTime start = createdTimeStart == null ? LocalDateTime.now().minusDays(7) : createdTimeStart;
164        LocalDateTime end = createdTimeEnd == null ? LocalDateTime.now() : createdTimeEnd;
165        Assert.isTrue(ChronoUnit.SECONDS.between(start, end) >= 0, "Store request constraint createdTimeStart must be earlier or equal to createdTimeEnd!");
166        if (noAudit) {
167            return ResponseEntity.ok(entityService.findAllByType(type));
168        } else {
169            return ResponseEntity.ok(entityService.findByCreatedTime(type, start, end, request.getParameterMap()));
170        }
171    }
172
173    @PreAuthorize("hasAuthority('read:type')")
174    @GetMapping("/api/v1/type")
175    public ResponseEntity<?> fetchOne(
176            @RequestParam(value = "oid", required = true) Long oid) {
177        return ResponseEntity.ok(entityService.findOne(oid));
178    }
