1   package com.fns.grivet.service;
2   
3   import com.codahale.metrics.MetricRegistry;
4   
5   import org.json.JSONObject;
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   import org.springframework.cloud.stream.annotation.StreamListener;
9   import org.springframework.cloud.stream.messaging.Sink;
10  import org.springframework.messaging.Message;
11  import org.springframework.util.Assert;
12  
13  public class PersistenceService {
14      
15      private final Logger log = LoggerFactory.getLogger(getClass());
16  
17      private final EntityService entityService;
18      private final MetricRegistry metricRegistry;
19  
20      public PersistenceService(EntityService entityService, MetricRegistry metricRegistry) {
21          this.entityService = entityService;
22          this.metricRegistry = metricRegistry;
23      }
24  
25      @StreamListener(Sink.INPUT)
26      public void store(Message<JSONObject> message) {
27          Assert.notNull(message.getHeaders(), "No message headers!");
28          String type = message.getHeaders().get("type", String.class);
29          Assert.hasText(type, "Type must not be null or empty!");
30          Assert.notNull(message.getPayload(), "Message must have non-null payload!");
31          log.debug("Received message.  Headers - {}.  Payload - {}", message.getHeaders().toString(),
32                  message.getPayload().toString());
33          entityService.create(type, message.getPayload());
34          metricRegistry.counter(MetricRegistry.name("store", type, "count")).inc();
35          log.info("Successfully stored type [{}]", type);
36      }
37  }
