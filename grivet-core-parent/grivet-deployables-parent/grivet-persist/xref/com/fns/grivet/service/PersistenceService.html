1   package com.fns.grivet.service;
2   
3   import org.json.JSONObject;
4   import org.springframework.cloud.stream.messaging.Processor;
5   import org.springframework.integration.MessageRejectedException;
6   import org.springframework.integration.annotation.ServiceActivator;
7   import org.springframework.integration.support.MessageBuilder;
8   import org.springframework.messaging.Message;
9   import org.springframework.util.Assert;
10  
11  import com.codahale.metrics.MetricRegistry;
12  import com.fns.grivet.model.Op;
13  
14  import lombok.extern.slf4j.Slf4j;
15  
16  @Slf4j
17  public class PersistenceService {
18  
19  	private final EntityService entityService;
20  	private final MetricRegistry metricRegistry;
21  
22  	public PersistenceService(EntityService entityService, MetricRegistry metricRegistry) {
23  		this.entityService = entityService;
24  		this.metricRegistry = metricRegistry;
25  	}
26  
27  	@ServiceActivator(inputChannel = Processor.INPUT, outputChannel = Processor.OUTPUT)
28  	public Object store(Message<JSONObject> message) {
29  		log.trace("Received message.  Headers - {}.  Payload - {}", message.getHeaders().toString(),
30  				message.getPayload().toString());
31  		Object result = null;
32  		try {
33  			Assert.notNull(message.getHeaders(), "No message headers!");
34  			Assert.notNull(message.getPayload(), "Message must have non-null payload!");
35  	
36  			Op op = message.getHeaders().get("op", Op.class);
37  			Assert.notNull(op, "Message header must contain an op code!");
38  	
39  			String type = null;
40  			Long oid = null;
41  		
42  			switch (op) {
43  				case CREATE:
44  					type = message.getHeaders().get("type", String.class);
45  					Assert.hasText(type, "Message header must contain a type for create requests!");
46  					entityService.create(type, message.getPayload());
47  					metricRegistry.counter(MetricRegistry.name("store", "create", type, "count")).inc();
48  					log.info("Successfully created type [{}]", type);
49  					break;
50  				case UPDATE:
51  					oid = message.getHeaders().get("oid", Long.class);
52  					Assert.notNull(oid, "Message header must contain an oid for update requests!");
53  					type = entityService.update(oid, message.getPayload());
54  					metricRegistry.counter(MetricRegistry.name("store", "update", type, "count")).inc();
55  					log.info("Successfully updated type [{}]", type);
56  					break;
57  				case DELETE:
58  					oid = message.getHeaders().get("oid", Long.class);
59  					Assert.notNull(oid, "Message header must contain an oid for delete requests!");
60  					type = entityService.delete(oid);
61  					metricRegistry.counter(MetricRegistry.name("store", "delete", type, "count")).inc();
62  					log.info("Successfully deleted type [{}]", type);
63  					break;
64  				default:
65  					throw new MessageRejectedException(message, "Bad payload!");
66  			}
67  			result = MessageBuilder.fromMessage(message).setHeader("processed", true);
68  			
69  		} catch (Exception e) {
70  			log.error("Op not available.  Headers - {}.  Payload - {}", message.getHeaders().toString(),
71  					message.getPayload().toString());
72  			result = MessageBuilder.fromMessage(message)
73  						.setHeader("processed", false)
74  						.setHeader("exception", e.getMessage());
75  		}
76  		return result;
77  	}
78  }
