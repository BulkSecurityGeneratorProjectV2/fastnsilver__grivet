074    public ResponseEntity<?> ingestCreateTypesRequest(@RequestHeader("Type") String type, @RequestBody JSONArray array) {
075        int numberOfTypesToCreate = array.length();
076        Assert.isTrue(numberOfTypesToCreate <= batchSize, String.format("The total number of entries in a request must not exceed %d! Your ingest request contained [%d] entries.", batchSize, numberOfTypesToCreate));
077        JSONObject payload = null;
078        HttpHeaders headers = new HttpHeaders();
079        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
080        for (int i = 0; i < numberOfTypesToCreate; i++) {
081            try {
082                payload = array.getJSONObject(i);
083                ingestService.ingest(MessageBuilder.withPayload(payload).setHeader("type", type).setHeader("op", Op.CREATE).build());
084                metricRegistry.counter(MetricRegistry.name("ingest", "create", type, "count")).inc();
085                log.info("Successfully ingested create request for type [{}]", type);
086            } catch (Exception e) {
087                String message = LogUtil.toLog(payload, String.format("Problems ingesting type! Portion of payload @ index[%d]\n", i + 1));
088                log.error(message, e);
089                if (numberOfTypesToCreate == 1) {
090                    throw e;
091                }
092                headers.set(String.format("Error[%s]", String.valueOf(i + 1)), e.getMessage());
093            }
094        }
095        return ResponseEntity.accepted().headers(headers).build();
096    }
