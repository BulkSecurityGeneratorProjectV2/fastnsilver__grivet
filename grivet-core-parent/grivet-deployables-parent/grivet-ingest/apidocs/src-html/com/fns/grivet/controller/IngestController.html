016package com.fns.grivet.controller;
017
018import org.json.JSONArray;
019import org.json.JSONObject;
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022import org.springframework.beans.factory.annotation.Autowired;
023import org.springframework.beans.factory.annotation.Value;
024import org.springframework.http.HttpHeaders;
025import org.springframework.http.MediaType;
026import org.springframework.http.ResponseEntity;
027import org.springframework.messaging.support.MessageBuilder;
028import org.springframework.security.access.prepost.PreAuthorize;
029import org.springframework.util.Assert;
030import org.springframework.web.bind.annotation.DeleteMapping;
031import org.springframework.web.bind.annotation.PatchMapping;
032import org.springframework.web.bind.annotation.PostMapping;
033import org.springframework.web.bind.annotation.RequestBody;
034import org.springframework.web.bind.annotation.RequestHeader;
035import org.springframework.web.bind.annotation.RequestMapping;
036import org.springframework.web.bind.annotation.RequestParam;
037import org.springframework.web.bind.annotation.RestController;
038
039import com.codahale.metrics.MetricRegistry;
040import com.fns.grivet.model.Op;
041import com.fns.grivet.service.Ingester;
042
043
044/**
045 * Ingestion end-points
046 *
047 * @author Chris Phillipson
048 */
049@RestController
050@RequestMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
051public class IngestController {
052
053    private final Logger log = LoggerFactory.getLogger(getClass());
054
055    @Value("${grivet.ingest.batch-size:100}")
056    private int batchSize;
057
058    private final Ingester ingestService;
059
060    private final MetricRegistry metricRegistry;
061
062    @Autowired
063    public IngestController(Ingester entityService, MetricRegistry metricRegistry) {
064        this.ingestService = entityService;
065        this.metricRegistry = metricRegistry;
066    }
067
068    @PreAuthorize("hasAuthority('write:type')")
069    @PostMapping("/api/v1/type")
070    public ResponseEntity<?> ingestCreateTypeRequest(@RequestHeader("Type") String type, @RequestBody JSONObject payload) {
071        ingestService
072                .ingest(MessageBuilder.withPayload(payload).setHeader("type", type).setHeader("op", Op.CREATE).build());
073        metricRegistry.counter(MetricRegistry.name("ingest", "create", type, "count")).inc();
074        log.info("Successfully ingested create request for type [{}]", type);
075        return ResponseEntity.accepted().build();
076    }
077    
078    @PreAuthorize("hasAuthority('write:type')")
079    @PostMapping("/api/v1/types")
080    public ResponseEntity<?> ingestCreateTypesRequest(@RequestHeader("Type") String type, @RequestBody JSONArray array) {
081        int numberOfTypesToCreate = array.length();
082        Assert.isTrue(numberOfTypesToCreate <= batchSize,
083                String.format(
084                        "The total number of entries in a request must not exceed %d! Your ingest request contained [%d] entries.",
085                        batchSize, numberOfTypesToCreate));
086        JSONObject payload = null;
087        HttpHeaders headers = new HttpHeaders();
088        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
089        for (int i = 0; i < numberOfTypesToCreate; i++) {
090            try {
091                payload = array.getJSONObject(i);
092                ingestService.ingest(MessageBuilder.withPayload(payload).setHeader("type", type)
093                        .setHeader("op", Op.CREATE).build());
094                metricRegistry.counter(MetricRegistry.name("ingest", "create", type, "count")).inc();
095                log.info("Successfully ingested create request for type [{}]", type);
096            } catch (Exception e) {
097                String message = LogUtil.toLog(payload,
098                        String.format("Problems ingesting type! Portion of payload @ index[%d]\n", i + 1));
099                log.error(message, e);
100                if (numberOfTypesToCreate == 1) {
101                    throw e;
102                }
103                headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
104            }
105        }
106        return ResponseEntity.accepted().headers(headers).build();
107    }
108
109    @PreAuthorize("hasAuthority('write:type')")
110    @PatchMapping("/api/v1/type")
111    public ResponseEntity<?> ingestUpdateTypeRequest(
112        @RequestParam(value = "oid", required = true) Long oid,
113        @RequestBody JSONObject payload) {
114        ingestService.ingest(MessageBuilder.withPayload(payload).setHeader("oid", oid).setHeader("op", Op.UPDATE).build());
115        metricRegistry.counter(MetricRegistry.name("ingest", "update", "count")).inc();
116        log.info("Successfully ingested update request for type w/ oid = [{}]", oid);
117        return ResponseEntity.accepted().build();
118    }
119    
120    @PreAuthorize("hasAuthority('delete:type')")
121    @DeleteMapping(value = "/api/v1/type")
122    public ResponseEntity<?> ingestDeleteTypeRequest(
123        @RequestParam(value = "oid", required = true) Long oid) {
124        ingestService.ingest(MessageBuilder.withPayload(new JSONObject()).setHeader("oid", oid).setHeader("op", Op.DELETE).build());
125        metricRegistry.counter(MetricRegistry.name("ingest", "delete", "count")).inc();
126        log.info("Successfully ingested update request for type w/ oid = [{}]", oid);
127        return ResponseEntity.accepted().build();
128    }
129
130}
