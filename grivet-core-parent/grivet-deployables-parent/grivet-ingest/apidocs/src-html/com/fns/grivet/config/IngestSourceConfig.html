001package com.fns.grivet.config;
002
003import com.fns.grivet.service.IngestService;
004import com.fns.grivet.service.Ingester;
005
006import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
007import org.springframework.cloud.stream.annotation.EnableBinding;
008import org.springframework.cloud.stream.messaging.Source;
009import org.springframework.context.annotation.Bean;
010import org.springframework.context.annotation.Configuration;
011import org.springframework.context.annotation.Primary;
012
013@Configuration
014public class IngestSourceConfig {
015
016    @ConditionalOnProperty(prefix = "app.ingest.source", name = "enabled", havingValue = "true")
017    @EnableBinding(Source.class)
018    static class EventSource {
019
020        @Primary
021        @Bean
022        public Ingester ingestService(Source source) {
023            return new IngestService(source);
024        }
025    }
026
027}
