071    @PreAuthorize("hasAuthority(\'write:typedef\')")
072    @PostMapping("/api/v1/definitions")
073    public ResponseEntity<?> defineTypes(@RequestBody JSONArray array) throws IOException, JSONException {
074        int numberOfTypesToRegister = array.length();
075        Assert.isTrue(numberOfTypesToRegister <= batchSize, String.format("The total number of entries in a type registration request must not exceed %d! Your registration request contained [%d] entries.", batchSize, numberOfTypesToRegister));
076        JSONObject payload = null;
077        String type = null;
078        HttpHeaders headers = new HttpHeaders();
079        URI location = null;
080        int errorCount = 0;
081        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
082        for (int i = 0; i < numberOfTypesToRegister; i++) {
083            try {
084                payload = array.getJSONObject(i);
085                type = classRegistryService.register(payload);
086                location = UriComponentsBuilder.newInstance().path("/api/v1/definition/{type}").buildAndExpand(type).toUri();
087                if (numberOfTypesToRegister == 1) {
088                    headers.setLocation(location);
089                } else {
090                    headers.set(String.format("Location[%s]", String.valueOf(i + 1)), location.toASCIIString());
091                }
092                log.info("Type [{}] successfully registered!", type);
093            } catch (Exception e) {
094                String message = LogUtil.toLog(payload, String.format("Problems registering type! Portion of payload @ index[%d]\n", i + 1));
095                log.error(message, e);
096                if (numberOfTypesToRegister == 1) {
097                    throw e;
098                }
099                headers.set(String.format("Error[%s]", String.valueOf(i + 1)), e.getMessage());
100                errorCount++;
101            }
102        }
103        return new ResponseEntity<>(headers, ((errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED));
104    }
