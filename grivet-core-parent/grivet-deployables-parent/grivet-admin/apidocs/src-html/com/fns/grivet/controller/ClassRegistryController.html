017
018import com.fns.grivet.service.ClassRegistryService;
019

022import org.json.JSONObject;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025import org.springframework.beans.factory.annotation.Autowired;
026import org.springframework.beans.factory.annotation.Value;

028import org.springframework.http.HttpStatus;

030import org.springframework.http.ResponseEntity;
031import org.springframework.security.access.prepost.PreAuthorize;

037import org.springframework.web.bind.annotation.RequestParam;

041import java.io.IOException;
042import java.net.URI;
043
044import io.swagger.annotations.Api;

046import io.swagger.annotations.ApiParam;

051/**
052 * Provides end-points for type registration and verification
053 * 
054 * @author Chris Phillipson
055 */
056@RestController
057@RequestMapping("/register")
058@Api(value = "register", produces = "application/json")
059public class ClassRegistryController {
060
061    private final Logger log = LoggerFactory.getLogger(getClass());
062    
063    @Value("${grivet.register.batch-size:100}")
064    private int batchSize;
065    
066    private final ClassRegistryService classRegistryService;
067    
068    @Autowired
069    public ClassRegistryController(ClassRegistryService classRegistryService) {
070        this.classRegistryService = classRegistryService;
071    }
072    
073    @PreAuthorize("hasRole(@roles.ADMIN)")
074    @RequestMapping(method = RequestMethod.POST,
075            consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
076    @ApiOperation(httpMethod = "POST", notes = "Register a type", value = "/register")
077    @ApiResponses({ @ApiResponse(code = 201, message = "Successfully registered type."),
078            @ApiResponse(code = 202, message = "Partial success. Location info for registered type(s). Error details for type(s) that could not be registered."),
079            @ApiResponse(code = 400, message = "Bad request."),
080            @ApiResponse(code = 500, message = "Internal server error.") })
081    public ResponseEntity<?> registerSingle(@RequestBody JSONObject json) throws IOException {
082        String type = classRegistryService.register(json);
083        UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
084        log.info("Type [{}] successfully registered!", type);
085        return ResponseEntity.created(ucb.path("/register/{type}").buildAndExpand(type).toUri()).build();
086    }
087
088    @PreAuthorize("hasRole(@roles.ADMIN)")
089    @RequestMapping(value = "/types", method = RequestMethod.POST, consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
090    @ApiOperation(httpMethod = "POST", notes = "Register multiple types", value = "/register/types")
091    @ApiResponses({ @ApiResponse(code = 201, message = "Successfully registered types."),
092            @ApiResponse(code = 202, message = "Partial success. Location info for registered type(s). Error details for type(s) that could not be registered."),
093            @ApiResponse(code = 400, message = "Bad request."),
094            @ApiResponse(code = 500, message = "Internal server error.") })
095    public ResponseEntity<?> registerMultiple(@RequestBody JSONArray json) throws IOException, JSONException {
096        int numberOfTypesToRegister = json.length();
097        Assert.isTrue(numberOfTypesToRegister <= batchSize,
098                String.format(
099                        "The total number of entries in a type registration request must not exceed %d! Your registration request contained [%d] entries.",
100                        batchSize, numberOfTypesToRegister));
101        JSONObject jsonObject = null;
102        String type = null;
103        HttpHeaders headers = new HttpHeaders();
104        URI location = null;
105        int errorCount = 0;
106        // allow for all JSONObjects within JSONArray to be processed; capture and report errors during processing
107        for (int i = 0; i < numberOfTypesToRegister; i++) {
108            try {
109                jsonObject = json.getJSONObject(i);
110                type = classRegistryService.register(jsonObject);
111                location = UriComponentsBuilder.newInstance().path("/register/{type}").buildAndExpand(type).toUri();
112                if (numberOfTypesToRegister == 1) {
113                    headers.setLocation(location); 
114                } else {
115                    headers.set(String.format("Location[%s]",String.valueOf(i+1)), location.toASCIIString());
116                }
117                log.info("Type [{}] successfully registered!", type);
118            } catch (Exception e) {
119                String message = LogUtil.toLog(jsonObject, String.format("Problems registering type! Portion of payload @ index[%d]\n", i+1));
120                log.error(message, e);
121                if (numberOfTypesToRegister == 1) {
122                    throw e;
123                }
124                headers.set(String.format("Error[%s]", String.valueOf(i+1)), e.getMessage());
125                errorCount++;
126            }
127        }
128        return new ResponseEntity<>(headers, ((errorCount == 0) ? HttpStatus.CREATED : HttpStatus.ACCEPTED));
129    }
130
131    @PreAuthorize("hasRole(@roles.ADMIN)")
132    @RequestMapping(value = "/{type}", method = RequestMethod.DELETE, produces = MediaType.APPLICATION_JSON_VALUE)
133    @ApiOperation(httpMethod = "DELETE", notes = "Delete the registered type.", value = "/register/{type}")
134    @ApiResponses({ @ApiResponse(code = 204, message = "Successfully deleted a registered type."),
135            @ApiResponse(code = 400, message = "Bad request."),
136            @ApiResponse(code = 500, message = "Internal server error.") })
137    public ResponseEntity<?> delete(
138            @ApiParam(value = "Type name", required = true)
139            @PathVariable("type") String type) {
140        classRegistryService.deregister(type);
141        log.info("Type [{}] successfully deregistered!", type);
142        return ResponseEntity.noContent().build();
143    }
144    
145    @PreAuthorize("hasRole(@roles.ADMIN) or hasRole(@roles.USER)")
146    @RequestMapping(value = "/{type}", produces = MediaType.APPLICATION_JSON_VALUE)
147    @ApiOperation(httpMethod = "GET", notes = "Retrieve the registered type.", value = "/register/{type}")
148    @ApiResponses({ @ApiResponse(code = 200, message = "Successfully retrieved a registered type."),
149            @ApiResponse(code = 400, message = "Bad request."),
150            @ApiResponse(code = 500, message = "Internal server error.") })
151    public ResponseEntity<?> get(
152            @ApiParam(value = "Type name", required = true)
153            @PathVariable("type") String type) {
154        JSONObject payload = classRegistryService.get(type);
155        String message = LogUtil.toLog(payload, String.format("Successfully retrieved type [%s]\n", type));
156        log.info(message);
157        return ResponseEntity.ok(payload.toString());
158    }
159    
160    @PreAuthorize("hasRole(@roles.ADMIN) or hasRole(@roles.USER)")
161    @RequestMapping(value = "", produces = MediaType.APPLICATION_JSON_VALUE)
162    @ApiOperation(httpMethod = "GET", notes = "All registered types.", value = "/register?showAll")
163    @ApiResponses({ @ApiResponse(code = 200, message = "List all registered types."),
164            @ApiResponse(code = 400, message = "Bad request."),
165            @ApiResponse(code = 500, message = "Internal server error.") })
166    public ResponseEntity<?> all(
167            @ApiParam(value = "Show all registered types?", required = true)
168            @RequestParam(value = "showAll", required = true) String showAll) {
169        return ResponseEntity.ok(classRegistryService.all().toString());
170    }
171        
172}
