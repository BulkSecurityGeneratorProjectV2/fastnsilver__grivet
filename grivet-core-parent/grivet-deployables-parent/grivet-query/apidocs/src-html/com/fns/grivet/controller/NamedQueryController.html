016package com.fns.grivet.controller;
017
018import org.apache.commons.lang3.StringUtils;
019import org.json.JSONArray;
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022import org.springframework.beans.factory.annotation.Autowired;
023import org.springframework.http.MediaType;
024import org.springframework.http.ResponseEntity;
025import org.springframework.security.access.prepost.PreAuthorize;
026import org.springframework.util.Assert;
027import org.springframework.util.MultiValueMap;
028import org.springframework.web.bind.annotation.DeleteMapping;
029import org.springframework.web.bind.annotation.GetMapping;
030import org.springframework.web.bind.annotation.PathVariable;
031import org.springframework.web.bind.annotation.PostMapping;
032import org.springframework.web.bind.annotation.RequestBody;
033import org.springframework.web.bind.annotation.RequestMapping;
034import org.springframework.web.bind.annotation.RequestParam;
035import org.springframework.web.bind.annotation.RestController;
036import org.springframework.web.util.UriComponentsBuilder;
037
038import com.fns.grivet.query.NamedQuery;
039import com.fns.grivet.query.QueryType;
040import com.fns.grivet.service.NamedQueryService;
041
042
043/**
044 * Provides end-points for registration, verification and execution of named queries
045 * 
046 * @author Chris Phillipson
047 */
048@RestController
049@RequestMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
050public class NamedQueryController {
051
052    private final Logger log = LoggerFactory.getLogger(getClass());
053    
054    private final NamedQueryService namedQueryService;
055    
056    @Autowired
057    public NamedQueryController(NamedQueryService namedQueryService) {
058        this.namedQueryService = namedQueryService;
059    }
060    
061    @PreAuthorize("hasAuthority('write:query')")
062    @PostMapping("/api/v1/query")
063    public ResponseEntity<?> createNamedQuery(
064            @RequestBody NamedQuery query) {
065        ResponseEntity<?> result = ResponseEntity.unprocessableEntity().build();
066        Assert.isTrue(StringUtils.isNotBlank(query.getName()), "Query name must not be null, empty or blank.");
067        Assert.notNull(query.getQuery(), "Query string must not be null!");
068        Assert.isTrue(isSupportedQuery(query), "Query must start with either CALL or SELECT");
069        // check for named parameters; but only if they exist
070        if (!query.getParams().isEmpty()) {
071            query.getParams().keySet()
072                .forEach(k -> Assert.isTrue(query.getQuery().contains(String.format(":%s", k)), String.format("Query must contain named parameter [%s]", k)));
073        }
074        namedQueryService.create(query);
075        log.info("Named Query \n\n{}\n\n successfully registered!", query);
076        UriComponentsBuilder ucb = UriComponentsBuilder.newInstance();
077        if (query.getParams().isEmpty()) {
078            result = ResponseEntity.created(ucb.path("/api/v1/query/{name}").buildAndExpand(query.getName()).toUri())
079                    .build();
080        } else {
081            result = ResponseEntity.noContent().build();
082        }
083        return result;
084    }
085    
086    @PreAuthorize("hasAuthority('execute:query')")
087    @GetMapping("/api/v1/query/{name}")
088    public ResponseEntity<?> executeNamedQuery(
089            @PathVariable("name") String name, 
090            @RequestParam MultiValueMap<String, ?> parameters) {
091        return ResponseEntity.ok(namedQueryService.get(name, parameters));
092    }
093    
094    @PreAuthorize("hasAuthority('list:query')")
095    @GetMapping("/api/v1/queries")
096    public ResponseEntity<?> listNamedQueries() {
097        JSONArray payload = namedQueryService.all();
098        return ResponseEntity.ok(payload.toString());
099    }
100    
101    @PreAuthorize("hasAuthority('delete:query')")
102    @DeleteMapping(value = "/api/v1/query/{name}")
103    public ResponseEntity<?> deleteNamedQuery(
104            @PathVariable("name") String name) {
105        namedQueryService.delete(name);
106        log.info("Query with name [{}] successfully deleted!", name);
107        return ResponseEntity.noContent().build();
108    }
109    
110    private boolean isSupportedQuery(NamedQuery query) {
111        boolean result = false;
112        if ((query.getQuery().toUpperCase().startsWith("SELECT") && query.getType().equals(QueryType.SELECT)) 
113                || (query.getQuery().toUpperCase().startsWith("CALL") && query.getType().equals(QueryType.SPROC))) {
114            result = true;
115        }
116        return result;
117    }
118}
