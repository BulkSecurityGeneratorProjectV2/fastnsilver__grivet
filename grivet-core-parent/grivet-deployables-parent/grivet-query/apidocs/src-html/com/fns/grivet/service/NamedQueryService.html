017
018import java.sql.CallableStatement;
019import java.sql.SQLException;
020import java.util.Iterator;
021import java.util.Map;
022import java.util.Set;
023
024import org.json.JSONArray;
025import org.json.JSONObject;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028import org.springframework.beans.factory.annotation.Autowired;
029import org.springframework.dao.DataAccessException;
030import org.springframework.jdbc.core.CallableStatementCallback;
031import org.springframework.jdbc.core.CallableStatementCreator;
032import org.springframework.jdbc.core.CallableStatementCreatorFactory;
033import org.springframework.jdbc.core.JdbcTemplate;
034import org.springframework.jdbc.core.SqlRowSetResultSetExtractor;
035import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
036import org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate;
037import org.springframework.jdbc.support.rowset.SqlRowSet;
038import org.springframework.stereotype.Service;
039import org.springframework.transaction.annotation.Transactional;
040import org.springframework.util.Assert;
041import org.springframework.util.CollectionUtils;
042import org.springframework.util.MultiValueMap;
043
044import com.fasterxml.jackson.databind.JsonNode;
045import com.fasterxml.jackson.databind.ObjectMapper;
046import com.fns.grivet.model.User;
047import com.fns.grivet.query.NamedQuery;
048import com.fns.grivet.repo.NamedQueryRepository;
049import com.fns.grivet.repo.SecurityFacade;
050
051@Service
052public class NamedQueryService {
053    
054    private final Logger log = LoggerFactory.getLogger(getClass());
055    
056    private final NamedQueryRepository namedQueryRepository;
057    private final JdbcTemplate jdbcTemplate;
058    private final NamedParameterJdbcTemplate namedParameterJdbcTemplate;
059    private final ObjectMapper mapper;
060    
061    @Autowired(required=false)
062    private SecurityFacade securityFacade;
063    
064    @Autowired
065    public NamedQueryService(NamedQueryRepository namedQueryRepository, JdbcTemplate jdbcTemplate, ObjectMapper mapper) {
066        this.namedQueryRepository = namedQueryRepository;
067        this.jdbcTemplate = jdbcTemplate;
068        this.namedParameterJdbcTemplate = new NamedParameterJdbcTemplate(jdbcTemplate);
069        this.mapper = mapper;
070    }
071    
072    @Transactional
073    public void create(NamedQuery namedQuery) {
074        User user = securityFacade != null ? securityFacade.getCurrentUser(): null;
075        namedQuery.setCreator(user);
076        namedQueryRepository.save(namedQuery);
077    }
078    
079    @Transactional(readOnly=true)
080    public String get(String name, MultiValueMap<String, ?> parameters) {
081        NamedQuery namedQuery = namedQueryRepository.findByName(name);
082        Assert.notNull(namedQuery, String.format("No query found that matches name [%s]!", name));
083        MapSqlParameterSource parameterSource = namedQuery.asParameterSource(parameters);
084        String sql = namedQuery.getQuery();
085        Map<String, String> sqlParams = namedQuery.getParams();
086        SqlRowSet rowSet = null;
087        if (parameterSource != null) {
088            // check that all params were supplied...
089            Set<String> queryParamKeys = sqlParams.keySet();
090            Set<String> parametersKeys = parameters.keySet();
091            Assert.isTrue(parametersKeys.containsAll(queryParamKeys), "Query cannot be executed! Missing query parameters!");
092            switch (namedQuery.getType()) {
093                case SELECT:
094                    rowSet = namedParameterJdbcTemplate.queryForRowSet(sql, parameterSource);
095                    break;
096                case SPROC:  
097                    String sproc = getProcedure(sql, queryParamKeys);
098                    CallableStatementCreatorFactory factory = new CallableStatementCreatorFactory(sproc, namedQuery.asSqlParameters(parameters));
099                    CallableStatementCreator csc = factory.newCallableStatementCreator(parameterSource.getValues());
100                    rowSet = callSproc(csc);
101                    break;
102            }
103        } else {
104            Assert.isTrue(sqlParams.isEmpty(), "Query cannot be executed! Query expects parameters!");
105            // ignore any supplied request params
106            switch (namedQuery.getType()) {
107                case SELECT:
108                    rowSet = namedParameterJdbcTemplate.getJdbcOperations().query(sql, new SqlRowSetResultSetExtractor());
109                    break;
110                case SPROC:
111                    String sproc = getProcedure(sql, null);
112                    CallableStatementCreatorFactory factory = new CallableStatementCreatorFactory(sproc, namedQuery.asSqlParameters(parameters));
113                    CallableStatementCreator csc = factory.newCallableStatementCreator((Map<String, ?>) null);
114                    rowSet = callSproc(csc);
115                    break;
116            }
117        }
118        return mapRows(rowSet);
119    }
120
121
122    @Transactional
123    public void delete(String name) {
124        NamedQuery nq = namedQueryRepository.findByName(name);
125        if (nq != null) {
126            namedQueryRepository.delete(nq);
127        }
128    }
129    
130    protected SqlRowSet callSproc(CallableStatementCreator csc) {
131       return jdbcTemplate.execute(csc, new CallableStatementCallback<SqlRowSet>() {
132            
133            @Override
134            public SqlRowSet doInCallableStatement(CallableStatement cs)
135                    throws SQLException, DataAccessException {
136                log.debug("Stored procedure to be executed: " + cs.toString());
137                return new SqlRowSetResultSetExtractor().extractData(cs.executeQuery());
138            }
139            
140        });
141    }
142    
143    protected String mapRows(SqlRowSet rowSet) {
144        JSONArray jsonArray = new JSONArray();
145        if (rowSet != null) {
146            String[] columnNames = rowSet.getMetaData().getColumnNames();
147            JSONObject jsonObject = null;
148            while(rowSet.next()) {
149                jsonObject = new JSONObject();
150                for (String columnName: columnNames) {
151                    jsonObject.put(columnName, rowSet.getObject(columnName));
152                }
153                jsonArray.put(jsonObject);
154            }
155        }
156        return jsonArray.toString();
157    }
158    
159    private String getProcedure(String sql, Set<String> queryParamKeys) {
160        if (!CollectionUtils.isEmpty(queryParamKeys)) {
161            for (String k: queryParamKeys) {
162               sql =  sql.replace(":" + k, "?");
163            }
164        }
165        if (!sql.startsWith("{")) {
166            sql = String.format("%s%s", "{", sql);
167        }
168        if (!sql.endsWith("}")) {
169            sql = String.format("%s%s", sql, "}");
170        }
171        return sql;
172    }
173    
174    @Transactional(readOnly=true)
175    public JSONArray all() {
176        JSONArray result = new JSONArray();
177        Iterable<NamedQuery> iterable = namedQueryRepository.findAll();
178        Assert.notNull(iterable, "No named queries are registered!");
179        Iterator<NamedQuery> it = iterable.iterator();
180        NamedQuery nq = null;
181        JSONObject jo = null;
182        JsonNode jn = null;
183        while(it.hasNext()) {
184           nq = it.next();
185           jn = mapper.valueToTree(nq);
186           jo = new JSONObject(jn.toString());
187           jo.remove("id");
188           jo.remove("createdTime");
189           result.put(jo);
190        }
191        return result;
192    }
193}
