001package com.fns.grivet.util;
002
003import java.sql.Timestamp;
004import java.time.LocalDateTime;
005
006import javax.persistence.AttributeConverter;
007import javax.persistence.Converter;
008
009@Converter(autoApply=true)
010public class LocalDateTimeConverter implements AttributeConverter<LocalDateTime, Timestamp> {
011
012    @Override
013    public Timestamp convertToDatabaseColumn(LocalDateTime attribute) {
014        return Timestamp.valueOf(attribute);
015    }
016
017    @Override
018    public LocalDateTime convertToEntityAttribute(Timestamp dbData) {
019        return dbData.toLocalDateTime();
020    }
021
022}
