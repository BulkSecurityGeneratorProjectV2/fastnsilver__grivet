16  package com.fns.grivet.service;
17  
18  import java.util.Iterator;
19  import java.util.List;
20  import java.util.Set;
21  
22  import org.json.JSONArray;
23  import org.json.JSONObject;
24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.springframework.dao.DuplicateKeyException;
26  import org.springframework.stereotype.Service;
27  import org.springframework.transaction.annotation.Transactional;
28  import org.springframework.util.Assert;
29  
30  import com.fns.grivet.model.Attribute;
31  import com.fns.grivet.model.AttributeType;
32  import com.fns.grivet.model.ClassAttribute;
33  import com.fns.grivet.model.User;
34  import com.fns.grivet.repo.AttributeRepository;
35  import com.fns.grivet.repo.AttributeTypeRepository;
36  import com.fns.grivet.repo.ClassAttributeRepository;
37  import com.fns.grivet.repo.ClassRepository;
38  
39  @Service
40  public class ClassRegistryService {
41  
42      private static final String TYPE = "type";
43      private static final String DESCRIPTION = "description";
44      private static final String ATTRIBUTES = "attributes";
45      
46      private final AttributeRepository attributeRepository;
47      private final AttributeTypeRepository attributeTypeRepository;
48      private final ClassRepository classRepository;
49      private final ClassAttributeRepository classAttributeRepository;
50      
51      @Autowired(required=false)
52      private SecurityFacade securityFacade;
53      
54      @Autowired
55      public ClassRegistryService(AttributeRepository attributeRepository,
56              AttributeTypeRepository attributeTypeRepository, ClassRepository classRepository,
57              ClassAttributeRepository classAttributeRepository) {
58          this.attributeRepository = attributeRepository;
59          this.attributeTypeRepository = attributeTypeRepository;
60          this.classRepository = classRepository;
61          this.classAttributeRepository = classAttributeRepository;
62      }
63  
64      @Transactional
65      // USE WITH CAUTION! Consider that w/ FK, type and all related data will be destroyed
66      public void deregister(String type) {
67          com.fns.grivet.model.Class c = classRepository.findByName(type);
68          if (c != null) {
69              classRepository.delete(c);
70          }
71      }
72      
73      @Transactional
74      public String register(JSONObject payload) {
75          String type = payload.getString(TYPE);
76          String description = payload.optString(DESCRIPTION);
77          com.fns.grivet.model.Class c = classRepository.findByName(type);
78          User user = securityFacade != null ? securityFacade.getCurrentUser(): null;
79          if (c == null) {
80              c = classRepository.save(new com.fns.grivet.model.Class(type, description, user));
81              JSONObject attributes = payload.getJSONObject(ATTRIBUTES);
82              Set<String> attributeNames = attributes.keySet();
83              Assert.notEmpty(attributeNames, String.format("[%s] must declare at least one attribute in registration request!", type));
84              Attribute a = null;
85              String at = null;
86              AttributeType attributeType = null;
87              // TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
88              for (String attributeName: attributeNames) {
89                  a = attributeRepository.findByName(attributeName);
90                  if (a == null) {
91                      a = attributeRepository.save(new Attribute(attributeName, user));
92                  }
93                  at = attributes.getString(attributeName);
94                  Assert.notNull(at, String.format("[%s].[%s] must declare an attribute type in registration request!", type, attributeName));
95                  attributeType = attributeTypeRepository.findByType(at);
96                  Assert.notNull(attributeType, String.format("Attribute type [%s] is not supported!", at));
97                  classAttributeRepository.save(new ClassAttribute(c.getId(), a.getId(), attributeType.getId(), user));
98              }
99          } else {
100             throw new DuplicateKeyException(String.format("Type [%s] already registered!", type));
101         }
102         return type;
103     }
104 
105     @Transactional(readOnly=true)
106     public JSONObject get(String type) {
107         com.fns.grivet.model.Class c = classRepository.findByName(type);
108         Assert.notNull(c, String.format("Type [%s] is not registered!", type));
109         JSONObject result = new JSONObject();
110         result.put("type", type);
111         String description = c.getDescription();
112         result.put("description", description != null ? description: "");
113         List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
114         Assert.notEmpty(cas, String.format("Type [%s] does not have any attributes registered!", type));
115         JSONObject attributes = new JSONObject();
116         result.put("attributes", attributes);
117         Attribute a = null;
118         AttributeType at = null;
119         for (ClassAttribute ca: cas) {
120             a = attributeRepository.findOne(ca.getAid());
121             Assert.notNull(a, String.format("Attribute id [%s] is not registered!", ca.getAid()));
122             at = attributeTypeRepository.findOne(ca.getTid());
123             Assert.notNull(a, String.format("Attribute Type id [%s] is not registered!", ca.getTid()));
124             attributes.put(a.getName(), at.getType());
125         }
126         return result;
127     }
128     
129     @Transactional(readOnly=true)
130     public JSONArray all() {
131         JSONArray result = new JSONArray();
132         Iterable<com.fns.grivet.model.Class> iterable = classRepository.findAll();
133         Assert.notNull(iterable, "No types are registered!");
134         Iterator<com.fns.grivet.model.Class> it = iterable.iterator();
135         com.fns.grivet.model.Class c = null;
136         while(it.hasNext()) {
137            c = it.next();
138            result.put(get(c.getName()));
139         }
140         return result;
141     }
