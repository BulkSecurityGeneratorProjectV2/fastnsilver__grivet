54      private final AttributeTypeRepository attributeTypeRepository;
55      private final EntityRepository entityRepository;
56      private final SchemaValidator schemaValidator;

58      @Autowired
59      public EntityService(ClassRepository classRepository, ClassAttributeRepository classAttributeRepository,
60              AttributeRepository attributeRepository, AttributeTypeRepository attributeTypeRepository,
61              EntityRepository entityRepository, SchemaValidator schemaValidator) {
62          this.classRepository = classRepository;
63          this.classAttributeRepository = classAttributeRepository;
64          this.attributeRepository = attributeRepository;
65          this.attributeTypeRepository = attributeTypeRepository;
66          this.entityRepository = entityRepository;
67          this.schemaValidator = schemaValidator;
68      }

70      @Transactional
71      public void create(String type, JSONObject payload) {
72          com.fns.grivet.model.Class c = classRepository.findByName(type);
73          Assert.notNull(c, String.format("Type [%s] is not registered!", type));
74          if (c.isValidatable()) {
75              ProcessingReport report = schemaValidator.validate(type, payload);
76              Assert.isTrue(report.isSuccess(), String.format("Cannot store [%s]! Type does not conform to JSON Schema definition.\n %s", type, report.toString()));
77          }
78          Set<String> keys = payload.keySet();
79          Assert.notEmpty(keys, String.format("Type [%s] must declare at least one attribute in create request!", type));
80          Object val = null;
81          Attribute a = null;
82          ClassAttribute ca = null;
83          AttributeType at = null;
84          Long eid = entityRepository.id(c.getId());
85          // TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
86          for (String key: keys) {
87              val = payload.get(key);
88              a = attributeRepository.findByName(key);
89              Assert.notNull(a, String.format("Attribute [%s] is not registered!", key));
90              ca = classAttributeRepository.findByCidAndAid(c.getId(), a.getId());
91              Assert.notNull(ca, String.format("[%s] is not a valid attribute of [%s]", key, type));
92              at = attributeTypeRepository.findOne(ca.getTid());
93              Assert.notNull(at, String.format("Attribute type [%s] is not supported!", at));
94              entityRepository.save(eid, a, at, val);
95          }
96      }
97      
98      @Transactional(readOnly=true)
99      public String findByCreatedTime(String type, LocalDateTime createdTimeStart, LocalDateTime createdTimeEnd, Set<Entry<String, String[]>> requestParameters) throws JsonProcessingException {
100         com.fns.grivet.model.Class c = classRepository.findByName(type);
101         Assert.notNull(c, "Type [%s] is not registered!");
102         List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
103         List<Attribute> attributes = Lists.newArrayList(attributeRepository.findAll());
104         Map<Integer, Integer> attributeToAttributeTypeMap = cas.stream().collect(Collectors.toMap(ClassAttribute::getAid, ClassAttribute::getTid));
105         Map<String, Integer> attributeNameToAttributeIdMap = attributes.stream().collect(Collectors.toMap(Attribute::getName, Attribute::getId));
106         DynamicQuery query = new DynamicQuery(requestParameters, attributeToAttributeTypeMap, attributeNameToAttributeIdMap);
107         List<EntityAttributeValue> rows = null;
108         if (query.hasConstraints()) {
109             rows = entityRepository.executeDynamicQuery(c.getId(), query);
110         } else {
111             rows = entityRepository.findByCreatedTime(c.getId(), createdTimeStart, createdTimeEnd);
112         }
113         return mapRows(attributeToAttributeTypeMap, rows);
114     }
115 
116     protected String mapRows(Map<Integer, Integer> attributeToAttributeTypeMap, List<EntityAttributeValue> rows) {
117         JSONArray jsonArray = new JSONArray();
118         Long current = null; 
119         Long previous = null;
120         JSONObject jsonObject = null;
121         for (EntityAttributeValue row: rows) {
122             current = row.getId();
123             if (!current.equals(previous)) {
124                 jsonObject = new JSONObject();
125                 jsonArray.put(jsonObject);
126             }
127             Integer tid = attributeToAttributeTypeMap.get(row.getAttributeId());
128             jsonObject.put(row.getAttributeName(), ValueHelper.getValue(attributeTypeRepository.findOne(tid), row.getAttributeValue()));
129             previous = current;
130         }
131         return jsonArray.toString();
132     }
