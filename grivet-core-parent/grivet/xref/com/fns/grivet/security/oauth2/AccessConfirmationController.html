16  package com.fns.grivet.security.oauth2;
17  
18  import java.security.Principal;
19  import java.util.LinkedHashMap;
20  import java.util.Map;
21  
22  import org.springframework.context.annotation.Profile;
23  import org.springframework.http.ResponseEntity;
24  import org.springframework.security.oauth2.common.util.OAuth2Utils;
25  import org.springframework.security.oauth2.provider.AuthorizationRequest;
26  import org.springframework.security.oauth2.provider.ClientDetails;
27  import org.springframework.security.oauth2.provider.ClientDetailsService;
28  import org.springframework.security.oauth2.provider.approval.Approval;
29  import org.springframework.security.oauth2.provider.approval.Approval.ApprovalStatus;
30  import org.springframework.security.oauth2.provider.approval.ApprovalStore;
31  import org.springframework.web.bind.annotation.RequestMapping;
32  import org.springframework.web.bind.annotation.RestController;
33  import org.springframework.web.bind.annotation.SessionAttributes;
34  
35  /**
36   * Controller for retrieving the model for and displaying the confirmation page for access to a protected resource.
37   * 
38   * @author Chris Phillipson
39   */
40  @Profile("oauth2")
41  @RestController
42  @SessionAttributes("authorizationRequest")
43  public class AccessConfirmationController {
44  
45      private ClientDetailsService clientDetailsService;
46  
47      private ApprovalStore approvalStore;
48  
49      @RequestMapping("/oauth/confirm_access")
50      public ResponseEntity<?> getAccessConfirmation(Map<String, Object> model, Principal principal) throws Exception {
51          AuthorizationRequest clientAuth = (AuthorizationRequest) model.remove("authorizationRequest");
52          ClientDetails client = clientDetailsService.loadClientByClientId(clientAuth.getClientId());
53          model.put("auth_request", clientAuth);
54          model.put("client", client);
55          Map<String, String> scopes = new LinkedHashMap<String, String>();
56          for (String scope : clientAuth.getScope()) {
57              scopes.put(OAuth2Utils.SCOPE_PREFIX + scope, "false");
58          }
59          for (Approval approval : approvalStore.getApprovals(principal.getName(), client.getClientId())) {
60              if (clientAuth.getScope().contains(approval.getScope())) {
61                  scopes.put(OAuth2Utils.SCOPE_PREFIX + approval.getScope(),
62                          approval.getStatus() == ApprovalStatus.APPROVED ? "true" : "false");
63              }
64          }
65          model.put("scopes", scopes);
66          return ResponseEntity.ok(model);
67      }
68  
69      @RequestMapping("/oauth/error")
70      public ResponseEntity<?> handleError(Map<String, Object> model) throws Exception {
71          // We can add more stuff to the model here for JSP rendering. If the client was a machine then
72          // the JSON will already have been rendered.
73          model.put("message", "There was a problem with the OAuth2 protocol");
74          return ResponseEntity.badRequest().body(model);
75      }
76  
77      public void setClientDetailsService(ClientDetailsService clientDetailsService) {
78          this.clientDetailsService = clientDetailsService;
79      }
80  
81      public void setApprovalStore(ApprovalStore approvalStore) {
82          this.approvalStore = approvalStore;
83      }
84  
85  }
