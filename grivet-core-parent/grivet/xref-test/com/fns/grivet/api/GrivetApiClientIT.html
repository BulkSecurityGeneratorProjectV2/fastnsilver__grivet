63      private String registerTestType() throws IOException {

139     @Test
140     public void testRegisterAndLinkAndStoreAndGetType_happyPath() throws IOException {
141         registerTestType();
142         Resource r = resolver.getResource("classpath:TestTypeSchema.json");
143         String schema = FileUtils.readFileToString(r.getFile());
144         given().contentType("application/json").request().body(schema).then().expect().statusCode(equalTo(200)).when().post("/type/schema/link");
145         r = resolver.getResource("classpath:TestTypeData.json");
146         String type = FileUtils.readFileToString(r.getFile());
147         given().contentType("application/json").request().body(type).then().expect().statusCode(equalTo(204)).when().post("/type/store/TestType");
148         
149         // GET (default)
150         Response response = given().contentType("application/json").request().then().expect().statusCode(equalTo(200)).when().get("/type/store/TestType");
151         JSONArray result = new JSONArray(response.body().asString());
152         JsonAssert.assertJsonEquals(type, result.get(0).toString());
153         
154         // GET (with equals constraint)
155         response = given().param("c", "varchar|equals|Rush").contentType("application/json").request().then().expect().statusCode(equalTo(200)).when().get("/type/store/TestType");
156         result = new JSONArray(response.body().asString());
157         JsonAssert.assertJsonEquals(type, result.get(0).toString());
158         
159         // GET (with startsWith constraint)
160         response = given().param("c", "text|startsWith|Grim-faced").contentType("application/json").request().then().expect().statusCode(equalTo(200)).when().get("/type/store/TestType");
161         result = new JSONArray(response.body().asString());
162         JsonAssert.assertJsonEquals(type, result.get(0).toString());
163         
164         // GET (with lessThanOrEqualTo constraint)
165         response = given().param("c", "datetime|lessThanOrEqualTo|2016-01-01T00:00:00Z").contentType("application/json").request().then().expect().statusCode(equalTo(200)).when().get("/type/store/TestType");
166         result = new JSONArray(response.body().asString());
167         JsonAssert.assertJsonEquals(type, result.get(0).toString());
168         
169         // GET (with multiple constraints)
170         response = given()
171                     .param("c", "datetime|lessThanOrEqualTo|2016-01-01T00:00:00Z")
172                     .param("c", "text|endsWith|city")
173                     .contentType("application/json").request().then().expect().statusCode(equalTo(200)).when().get("/type/store/TestType");
174         result = new JSONArray(response.body().asString());
175         JsonAssert.assertJsonEquals(type, result.get(0).toString());
176         
177         deregisterType();
178     }
