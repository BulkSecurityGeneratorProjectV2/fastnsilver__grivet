016package com.fns.grivet.query;
017
018import java.sql.Timestamp;
019import java.sql.Types;
020import java.time.Instant;
021import java.time.LocalDate;
022import java.time.LocalDateTime;
023import java.util.ArrayList;
024import java.util.Collections;
025import java.util.List;
026import java.util.Map;
027import java.util.Map.Entry;
028import java.util.Objects;
029import java.util.Set;
030import java.util.stream.Collectors;
031
032import org.springframework.jdbc.core.SqlParameterValue;
033import org.springframework.util.Assert;
034import org.springframework.util.StringUtils;
035
036import com.fns.grivet.model.AttributeType;
037
038
039public class DynamicQuery {
040
041    private final List<Constraint> constraints = new ArrayList<>();
042    private final Map<Integer, Integer> attributeToAttributeTypeMap;
043    private final Map<String, Integer> attributeNameToAttributeIdMap;
044    
045    public DynamicQuery(Set<Entry<String, String[]>> requestParams, 
046            Map<Integer, Integer> attributeToAttributeTypeMap, 
047            Map<String, Integer> attributeNameToAttributeIdMap) {
048        String k = null;
049        String[] v = null;
050        if (requestParams != null && !requestParams.isEmpty()) {
051            for (Entry<String, String[]> requestParam: requestParams) {
052                k = requestParam.getKey();
053                v = requestParam.getValue();
054                if (isConstraintKey(k)) {
055                    for (String cs: v) {
056                        if (isConstraintValue(cs)) {
057                            constraints.add(new Constraint(cs.split("\\|")));
058                        }
059                    }
060                }
061            }
062        }
063        this.attributeToAttributeTypeMap = attributeToAttributeTypeMap;
064        this.attributeNameToAttributeIdMap = attributeNameToAttributeIdMap;
065    }
066    
067    public boolean hasConstraints() {
068        return constraints.size() > 0;
069    }
070    
071    List<Constraint> getConstraints() {
072        return Collections.unmodifiableList(constraints);     
073    }
074    
075    public boolean containsAndConjunction() {
076        return constraints
077                .stream()
078                .filter(c -> Objects.equals(c.getConjunction(), Conjunction.AND))
079                .collect(Collectors.toList()).size() >= 1;
080    }
081    
082    public boolean areConjunctionsHomogenous() {
083        boolean result = true;
084        if (constraints.size() >=1) {
085            Conjunction current = null;
086            Conjunction prior = null;
087            for (Constraint c: constraints) {
088                current = c.getConjunction();
089                if (current!= null && prior != null && current != prior) {
090                    result = false;
091                    break;
092                }
093                prior = current;
094            }
095        }
096        return result;
097    }
098    
099    public SqlParameterValue[] asSqlParameterValues() {
100        List<SqlParameterValue> paramValues = new ArrayList<>();
101        Integer attributeId = null;
102        Integer attributeTypeId = null;
103        AttributeType at = null;
104        String attributeName = null;
105        for (Constraint c: constraints) {
106            attributeName = c.getAttributeName();
107            Assert.isTrue(attributeName !=null && !attributeName.isEmpty(), "Attribute name must not be null or empty!");
108            attributeId = attributeNameToAttributeIdMap.get(c.getAttributeName());
109            Assert.notNull(attributeId, String.format("Invalid query constraint! No attribute identifier found for [%s]", attributeName));
110            paramValues.add(new SqlParameterValue(Types.INTEGER, attributeId));
111            attributeTypeId = attributeToAttributeTypeMap.get(attributeId);
112            Assert.notNull(attributeTypeId, String.format("Invalid query constraint! No attribute type identifier found for attribute name [%s]", attributeName));
113            at = AttributeType.fromId(attributeTypeId);
114            Assert.notNull(attributeTypeId, String.format("Invalid query constraint! No matching attribute type found for attribute name [%s]", attributeName));
115            if (c.getOperator().equals(Operator.BETWEEN)) {
116                paramValues.add(getSqlParameterValue(at, c.getOperator(), c.getValues()[0]));
117                paramValues.add(getSqlParameterValue(at, c.getOperator(), c.getValues()[1]));
118            } else if (c.getOperator().equals(Operator.IN)) {
119                for (String value: c.getValues()) {
120                    paramValues.add(getSqlParameterValue(at, c.getOperator(), value));
121                }
122            } else {
123                paramValues.add(getSqlParameterValue(at, c.getOperator(), c.getValues()[0]));
124            }
125        }
126        return paramValues.toArray(new SqlParameterValue[paramValues.size()]);
127    }
128    
129    private boolean isConstraintKey(String key) {
130        boolean result = false;
131        if ((key.equalsIgnoreCase("c") || key.equalsIgnoreCase("constraint"))) {
132            result = true;
133        }
134        return result;
135    }
136    
137    private boolean isConstraintValue(String value) {
138        boolean result = false;
139        if (value != null && StringUtils.countOccurrencesOf(value,"|") >= 2) {
140            result = true;
141        }
142        return result;
143    }
144    
145    protected SqlParameterValue getSqlParameterValue(AttributeType at, Operator op, String value) {
146        SqlParameterValue result = null;
147        switch (at) {
148            case BIG_INTEGER:
149                result = new SqlParameterValue(at.getSqlType(), Long.valueOf(value));
150                break;
151            case ISO_DATE:
152                result = new SqlParameterValue(at.getSqlType(), java.sql.Date.valueOf(LocalDate.parse(value)));
153                break;
154            case ISO_DATETIME:
155                result = new SqlParameterValue(at.getSqlType(), Timestamp.valueOf(LocalDateTime.parse(value)));
156                break;
157            case ISO_INSTANT:
158                result = new SqlParameterValue(at.getSqlType(), Timestamp.from(Instant.parse(value)));
159                break;
160            case DECIMAL:
161                result = new SqlParameterValue(at.getSqlType(), Double.valueOf(value));
162                break;
163            case INTEGER:
164                result = new SqlParameterValue(at.getSqlType(), Integer.valueOf(value));
165                break;
166            case VARCHAR:
167                result = new SqlParameterValue(at.getSqlType(), getWildcardedValue(op, value));
168                break;
169            case TEXT:
170                result = new SqlParameterValue(at.getSqlType(), getWildcardedValue(op, value));
171                break;
172            case JSON_BLOB:
173                result = new SqlParameterValue(at.getSqlType(), getWildcardedValue(op, value));
174                break;
175        }
176        return result;
177    }
178    
179    protected String getWildcardedValue(Operator op, String value) {
180        String result = null;
181        switch (op) {
182            case STARTS_WITH:
183                result = String.format("%s%s", value, "%");
184                break;
185            case ENDS_WITH:
186                result = String.format("%s%s", "%", value);
187                break;
188            case CONTAINS:
189                result = String.format("%s%s%s", "%", value, "%");
190                break;
191            default:
192                result = value;
193                break;
194        }
195        return result;
196    }
197    
198}
