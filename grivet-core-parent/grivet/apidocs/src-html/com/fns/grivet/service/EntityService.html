048@Service
049public class EntityService {
050
051    private final ClassRepository classRepository;
052    private final ClassAttributeRepository classAttributeRepository;
053    private final AttributeRepository attributeRepository;
054    private final AttributeTypeRepository attributeTypeRepository;
055    private final EntityRepository entityRepository;
056    private final SchemaValidator schemaValidator;
057    
058    @Autowired
059    public EntityService(ClassRepository classRepository, ClassAttributeRepository classAttributeRepository,
060            AttributeRepository attributeRepository, AttributeTypeRepository attributeTypeRepository,
061            EntityRepository entityRepository, SchemaValidator schemaValidator) {
062        this.classRepository = classRepository;
063        this.classAttributeRepository = classAttributeRepository;
064        this.attributeRepository = attributeRepository;
065        this.attributeTypeRepository = attributeTypeRepository;
066        this.entityRepository = entityRepository;
067        this.schemaValidator = schemaValidator;
068    }
069
070    @Transactional
071    public void create(String type, JSONObject payload) {
072        com.fns.grivet.model.Class c = classRepository.findByName(type);
073        Assert.notNull(c, String.format("Type [%s] is not registered!", type));
074        if (c.isValidatable()) {
075            ProcessingReport report = schemaValidator.validate(type, payload);
076            Assert.isTrue(report.isSuccess(), String.format("Cannot store [%s]! Type does not conform to JSON Schema definition.\n %s", type, report.toString()));
077        }
078        Set<String> keys = payload.keySet();
079        Assert.notEmpty(keys, String.format("Type [%s] must declare at least one attribute in create request!", type));
080        Object val = null;
081        Attribute a = null;
082        ClassAttribute ca = null;
083        AttributeType at = null;
084        Long eid = entityRepository.id(c.getId());
085        // TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
086        for (String key: keys) {
087            val = payload.get(key);
088            a = attributeRepository.findByName(key);
089            Assert.notNull(a, String.format("Attribute [%s] is not registered!", key));
090            ca = classAttributeRepository.findByCidAndAid(c.getId(), a.getId());
091            Assert.notNull(ca, String.format("[%s] is not a valid attribute of [%s]", key, type));
092            at = attributeTypeRepository.findOne(ca.getTid());
093            Assert.notNull(at, String.format("Attribute type [%s] is not supported!", at));
094            entityRepository.save(eid, a, at, val);
095        }
096    }
097    
098    @Transactional(readOnly=true)
099    public String findByCreatedTime(String type, LocalDateTime createdTimeStart, LocalDateTime createdTimeEnd, Set<Entry<String, String[]>> requestParameters) throws JsonProcessingException {
100        com.fns.grivet.model.Class c = classRepository.findByName(type);
101        Assert.notNull(c, "Type [%s] is not registered!");
102        List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
103        List<Attribute> attributes = Lists.newArrayList(attributeRepository.findAll());
104        Map<Integer, Integer> attributeToAttributeTypeMap = cas.stream().collect(Collectors.toMap(ClassAttribute::getAid, ClassAttribute::getTid));
105        Map<String, Integer> attributeNameToAttributeIdMap = attributes.stream().collect(Collectors.toMap(Attribute::getName, Attribute::getId));
106        DynamicQuery query = new DynamicQuery(requestParameters, attributeToAttributeTypeMap, attributeNameToAttributeIdMap);
107        List<EntityAttributeValue> rows = null;
108        if (query.hasConstraints()) {
109            rows = entityRepository.executeDynamicQuery(c.getId(), query);
110        } else {
111            rows = entityRepository.findByCreatedTime(c.getId(), createdTimeStart, createdTimeEnd);
112        }
113        return mapRows(attributeToAttributeTypeMap, rows);
114    }
115
116    protected String mapRows(Map<Integer, Integer> attributeToAttributeTypeMap, List<EntityAttributeValue> rows) {
117        JSONArray jsonArray = new JSONArray();
118        Long current = null; 
119        Long previous = null;
120        JSONObject jsonObject = null;
121        for (EntityAttributeValue row: rows) {
122            current = row.getId();
123            if (!current.equals(previous)) {
124                jsonObject = new JSONObject();
125                jsonArray.put(jsonObject);
126            }
127            Integer tid = attributeToAttributeTypeMap.get(row.getAttributeId());
128            jsonObject.put(row.getAttributeName(), ValueHelper.getValue(attributeTypeRepository.findOne(tid), row.getAttributeValue()));
129            previous = current;
130        }
131        return jsonArray.toString();
132    }
