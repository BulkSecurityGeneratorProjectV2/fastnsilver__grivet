001package com.fns.grivet.config;
002
003import org.springframework.beans.factory.annotation.Value;
004import org.springframework.context.annotation.Bean;
005import org.springframework.context.annotation.Configuration;
006
007import springfox.documentation.builders.PathSelectors;
008import springfox.documentation.builders.RequestHandlerSelectors;
009import springfox.documentation.service.ApiInfo;
010import springfox.documentation.service.Contact;
011import springfox.documentation.spi.DocumentationType;
012import springfox.documentation.spring.web.plugins.Docket;
013import springfox.documentation.swagger2.annotations.EnableSwagger2;
014
015@Configuration
016@EnableSwagger2
017public class SwaggerConfig {
018
019    @Bean
020    public Docket api(
021            @Value("${info.title}") String projectName,
022            @Value("${info.description}") String projectDescription) { 
023        return new Docket(DocumentationType.SWAGGER_2)  
024          .select()                                  
025          .apis(RequestHandlerSelectors.basePackage("com.fns.grivet.controller"))             
026          .paths(PathSelectors.any())
027          .build()                                          
028              .apiInfo(apiInfo(projectName, projectDescription));
029    }
030    
031    private ApiInfo apiInfo(
032            String projectName,
033            String projectDescription) {
034        ApiInfo apiInfo = new ApiInfo(
035          String.format("%s REST API", projectName),
036          projectDescription,
037          "v1",
038          "https://www.apache.org/licenses/LICENSE-2.0",
039          new Contact("Chris Phillipson", "http://techblitz.io/grivet/", "fastnsilver@gmail.com"),
040          "Apache License 2.0",
041          "https://www.apache.org/licenses/LICENSE-2.0");
042        return apiInfo;
043    }
044}
