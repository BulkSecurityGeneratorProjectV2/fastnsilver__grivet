001package com.fns.grivet.config;
002
003import org.springframework.boot.autoconfigure.security.SecurityProperties;
004import org.springframework.context.annotation.Bean;
005import org.springframework.context.annotation.Configuration;
006import org.springframework.context.annotation.Profile;
007import org.springframework.core.annotation.Order;
008import org.springframework.security.config.annotation.method.configuration.EnableGlobalMethodSecurity;
009import org.springframework.security.config.annotation.web.builders.HttpSecurity;
010import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
011
012import com.auth0.Auth0;
013import com.auth0.authentication.AuthenticationAPIClient;
014import com.auth0.authentication.result.UserProfile;
015import com.auth0.request.Request;
016import com.auth0.spring.security.api.Auth0JWTToken;
017import com.auth0.spring.security.api.Auth0SecurityConfig;
018
019@Profile("secure")
020@Configuration
021@EnableWebSecurity(debug = true)
022@EnableGlobalMethodSecurity(prePostEnabled = true)
023@Order(SecurityProperties.ACCESS_OVERRIDE_ORDER)
024public class SpringSecurityWebAppConfig extends Auth0SecurityConfig {
025
026    public static class Auth0Client {
027
028        private final Auth0 auth0;
029        private final AuthenticationAPIClient client;
030
031        public Auth0Client(String clientid, String domain) {
032            this.auth0 = new Auth0(clientid, domain);
033            this.client = this.auth0.newAuthenticationAPIClient();
034        }
035
036        public String getUsername(Auth0JWTToken token) {
037            final Request<UserProfile> request = client.tokenInfo(token.getJwt());
038            final UserProfile profile = request.execute();
039            return profile.getEmail();
040        }
041
042    }
043    
044    /**
045     * Not required for the Spring Security implementation, but offers Auth0 API access
046     */
047    @Bean
048    public Auth0Client auth0Client() {
049        return new Auth0Client(clientId, issuer);
050    }
051
052    /**
053     * Override this function in subclass to apply custom authentication / authorization
054     * strategies to your application endpoints
055     */
056    @Override
057    protected void authorizeRequests(final HttpSecurity http) throws Exception {
058        http.authorizeRequests()
059                .antMatchers(securedRoute).authenticated();
060    }
061
062    
063    
064}
