017package com.fns.grivet.query;
018
019import com.fasterxml.jackson.annotation.JsonCreator;
020import com.fasterxml.jackson.annotation.JsonIgnore;
021import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
022import com.fasterxml.jackson.annotation.JsonProperty;
023import com.fasterxml.jackson.annotation.JsonPropertyOrder;
024import com.fns.grivet.model.AttributeType;
025import com.fns.grivet.model.Audited;
026
027import org.apache.commons.lang3.builder.EqualsBuilder;
028import org.apache.commons.lang3.builder.HashCodeBuilder;
029import org.apache.commons.lang3.builder.ToStringBuilder;
030import org.springframework.jdbc.core.SqlParameter;
031import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
032import org.springframework.util.CollectionUtils;
033import org.springframework.util.MultiValueMap;
034
035import java.util.ArrayList;
036import java.util.HashMap;
037import java.util.List;
038import java.util.Map;
039import java.util.Map.Entry;
040import java.util.Set;
041
042import javax.persistence.CollectionTable;
043import javax.persistence.Column;
044import javax.persistence.ElementCollection;
045import javax.persistence.Entity;
046import javax.persistence.EnumType;
047import javax.persistence.Enumerated;
048import javax.persistence.FetchType;
049import javax.persistence.GeneratedValue;
050import javax.persistence.Id;
051import javax.persistence.JoinColumn;
052import javax.persistence.MapKeyColumn;
053import javax.validation.Valid;
054import javax.validation.constraints.Size;
055
056@JsonIgnoreProperties(ignoreUnknown = true)
057@JsonPropertyOrder({
058    "name",
059    "type",
060    "query",
061    "params"
062})
063@Entity
064public class NamedQuery extends Audited {
065
066    private static final long serialVersionUID = 1L;
067    
068    @JsonIgnore
069    @Id
070    @GeneratedValue
071    private Integer id;
072    
073    @JsonProperty("name")
074    @Size(max=255)
075    @Column(nullable=false, unique=true)
076    private String name;
077    
078    @JsonProperty("type")
079    @Enumerated(EnumType.STRING)
080    private QueryType type;
081    
082    @JsonProperty("query")
083    @Size(max=2000)
084    @Column(length=2000, nullable=false)
085    private String query;
086    
087    @JsonProperty("params")
088    @Valid
089    @ElementCollection(fetch=FetchType.EAGER)
090    @CollectionTable(name="named_query_parameter", joinColumns=@JoinColumn(name="id"))
091    @MapKeyColumn(name="parameter_name")
092    @Column(name="parameter_type", nullable=false)
093    private Map<String, String> params;
094    
095    protected NamedQuery() {
096        super();
097    }
098    
099    @JsonCreator
100    public NamedQuery(@JsonProperty("name") String name, @JsonProperty("type") QueryType type, @JsonProperty("query") String query, @JsonProperty("params") Map<String, String> params) {
101        this.name = name;
102        this.query = query;
103        setType(type);
104        this.params = params;
105    }
106    
107    public Integer getId() {
108        return id;
109    }
110
111    public String getName() {
112        return name;
113    }
114    
115    private void setType(QueryType type) {
116       if (type == null) {
117           if (query.toUpperCase().startsWith("SELECT")) {
118               this.type = QueryType.SELECT;
119           }
120           if (query.toUpperCase().startsWith("CALL")) {
121               this.type = QueryType.SPROC;
122           }
123       } else {
124           this.type = type;
125       }
126    }
127    
128    public QueryType getType() {
129        return type;
130    }
131
132    public String getQuery() {
133        return query;
134    }
135
136    public Map<String, String> getParams() {
137        if (params == null) {
138            params = new HashMap<String, String>();
139        }
140        return params;
141    }
142    
143    // only when parameter values are not null, empty, or blank 
144    // and parameter definition for named query is not empty
145    // will a MapSqlParameterSource be constructed
146    public MapSqlParameterSource asParameterSource(MultiValueMap<String, ?> parameterValues) {
147        MapSqlParameterSource paramSource = null;
148        if (!CollectionUtils.isEmpty(parameterValues) && !getParams().isEmpty()) {
149            paramSource = new MapSqlParameterSource();
150            Set<Entry<String, String>> paramNameToParamTypeEntries = getParams().entrySet();
151            List<?> values = null; 
152            for (Entry<String, String> entry: paramNameToParamTypeEntries) {
153                values = parameterValues.get(entry.getKey());
154                if (!CollectionUtils.isEmpty(values)) {
155                    if (values.size() == 1) {
156                        paramSource.addValue(entry.getKey(), parameterValues.getFirst(entry.getKey()), AttributeType.toSqlType(entry.getValue()));
157                    } else {
158                        paramSource.addValue(entry.getKey(), values);
159                    }
160                }
161            }
162        }
163        return paramSource;
164    }
165    
166    // only when parameter values are not null, empty, or blank 
167    // and parameter definition for named query is not empty
168    // will a List<SqlParameter> be constructed
169    public List<SqlParameter> asSqlParameters(MultiValueMap<String, ?> parameterValues) {
170        List<SqlParameter> sqlParams = null;
171        if (!CollectionUtils.isEmpty(parameterValues) && !getParams().isEmpty()) {
172            sqlParams = new ArrayList<>();
173            Set<Entry<String, String>> paramNameToParamTypeEntries = getParams().entrySet();
174            List<?> values = null; 
175            for (Entry<String, String> entry: paramNameToParamTypeEntries) {
176                values = parameterValues.get(entry.getKey());
177                if (!CollectionUtils.isEmpty(values)) {
178                    if (values.size() == 1) {
179                        sqlParams.add(new SqlParameter(entry.getKey(), AttributeType.toSqlType(entry.getValue())));
180                    } else {
181                        throw new IllegalArgumentException(String.format("Stored Procedure cannot be executed! Parameter [%s] is not a scalar value!", entry.getKey()));
182                    }
183                }
184            }
185        }
186        return sqlParams;
187    }
188    
189    @Override
190    public int hashCode() {
191        return HashCodeBuilder.reflectionHashCode(this);
192    }
193
194    @Override
195    public boolean equals(Object object) {
196        return EqualsBuilder.reflectionEquals(this, object);
197    }
198
199    @Override
200    public String toString() {
201        return ToStringBuilder.reflectionToString(this);
202    }
203}
