060
061@Profile("oauth2")
062@Configuration
063public class SecurityConfig {
064    
065    @Profile("oauth2")
066    @Configuration
067    @EnableResourceServer
068    protected static class ResourceServer extends
069            ResourceServerConfigurerAdapter {
070
071        @Autowired
072        private TokenStore tokenStore;
073
074        @Override
075        public void configure(ResourceServerSecurityConfigurer resources) throws Exception {
076            resources
077                .tokenStore(tokenStore)
078                .stateless(false);
079        }
080
081        @Override
082        public void configure(HttpSecurity http) throws Exception {
083            http
084                // Since we want the protected resources to be accessible in the UI as well we need 
085                // session creation to be allowed (it's disabled by default in 2.0.6)
086                .sessionManagement().sessionCreationPolicy(SessionCreationPolicy.IF_REQUIRED)
087            .and()
088                .requestMatchers().antMatchers("/oauth/users/**", "/oauth/clients/**")
089            .and()
090            // default protection for all resources (including /oauth/authorize)
091                .authorizeRequests()
092                    .antMatchers(HttpMethod.GET, "/users").hasRole("ADMIN")
093                    .anyRequest().hasRole("USER");
094            // ... more configuration
095        }
096
097    }
098
099    @Profile("oauth2")
100    @Configuration
101    @EnableAuthorizationServer
102    protected static class OAuth2Config extends
103            AuthorizationServerConfigurerAdapter {
104
105        @Value("${oauth2.resource-id:grivet}")
106        private String resourceId;
107        
108        @Value("${oauth2.access-token-validity-seconds:60}")
109        private int accessTokenValiditySeconds;
110        
111        @Value("${oauth2.refresh-token-validity-seconds:600}")
112        private int refreshTokenValiditySeconds;
113        
114        @Autowired
115        private CustomUserDetailsService userDetailsService;
116        
117        @Autowired
118        @Qualifier("authenticationManagerBean")
119        private AuthenticationManager authenticationManager;
120
121        @Autowired
122        private DataSource dataSource;
123        
124        @Autowired
125        private UserApprovalHandler userApprovalHandler;
126
127        private BCryptPasswordEncoder passwordEncoder = new BCryptPasswordEncoder();
128
129        @Bean
130        public JdbcTokenStore tokenStore() {
131            return new JdbcTokenStore(dataSource);
132        }
133
134        @Bean
135        protected AuthorizationCodeServices authorizationCodeServices() {
136            return new JdbcAuthorizationCodeServices(dataSource);
137        }
138        
139        @Override
140        public void configure(AuthorizationServerSecurityConfigurer authServer) throws Exception {
141            authServer.passwordEncoder(passwordEncoder);
142            authServer.allowFormAuthenticationForClients();
143        }
144
145        @Override
146        public void configure(AuthorizationServerEndpointsConfigurer endpoints) throws Exception {
147            endpoints
148                .authorizationCodeServices(authorizationCodeServices())
149                .authenticationManager(authenticationManager)
150                .userApprovalHandler(userApprovalHandler)
151                .tokenStore(tokenStore())
152                .userDetailsService(userDetailsService);
153        }
154
155        @Override
156        public void configure(ClientDetailsServiceConfigurer clients) throws Exception {
157            // @formatter:off
158            clients.jdbc(dataSource)
159                .passwordEncoder(passwordEncoder)
160                .withClient("admin-client")
161                    .authorizedGrantTypes("password", "authorization_code", "refresh_token", "implicit")
162                    .authorities("ROLE_ADMIN", "ROLE_USER")
163                    .scopes("read", "write", "trust")
164                    .resourceIds(resourceId)
165                    .accessTokenValiditySeconds(accessTokenValiditySeconds)
166                    .refreshTokenValiditySeconds(refreshTokenValiditySeconds);
167            // @formatter:on
168        }
169
170    }
171    
172    @Profile("oauth2")
173    @Configuration
174    protected static class ApprovalStoreConfig {
175        
176        @Autowired
177        private ClientDetailsService clientDetailsService;
178        
179        @Autowired
180        private TokenStore tokenStore;
181        
182        @Bean
183        public ApprovalStore approvalStore() throws Exception {
184            TokenApprovalStore store = new TokenApprovalStore();
185            store.setTokenStore(tokenStore);
186            return store;
187        }
188        
189        @Bean
190        @Lazy
191        @Scope(proxyMode = ScopedProxyMode.TARGET_CLASS)
192        public GrivetUserApprovalHandler userApprovalHandler() throws Exception {
193            GrivetUserApprovalHandler handler = new GrivetUserApprovalHandler();
194            handler.setApprovalStore(approvalStore());
195            handler.setRequestFactory(new DefaultOAuth2RequestFactory(clientDetailsService));
196            handler.setClientDetailsService(clientDetailsService);
197            handler.setUseApprovalStore(true);
198            return handler;
199        }
200    }
201    
202    @Profile("oauth2")
203    @Configuration
204    @EnableWebSecurity
205    protected static class WebSecurityConfig extends WebSecurityConfigurerAdapter {
206        
207        @Override
208        @Bean
209        public AuthenticationManager authenticationManagerBean() throws Exception {
210            return super.authenticationManagerBean();
211        }
212        
213        @Override
214        public void configure(WebSecurity web) throws Exception {
215            web.ignoring().antMatchers("/webjars/**", "/images/**", "/oauth/uncache_approvals", "/oauth/cache_approvals");
216        }
217    }
218    
219    
220    @Profile("oauth2")
221    @Configuration
222    @EnableGlobalMethodSecurity(prePostEnabled = true)
223    protected static class MethodSecurityConfig extends GlobalMethodSecurityConfiguration {
224        
225        @Override
226        protected MethodSecurityExpressionHandler createExpressionHandler() {
227            return new OAuth2MethodSecurityExpressionHandler();
228        }
229    }
