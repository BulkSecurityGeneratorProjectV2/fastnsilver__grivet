001package com.fns.grivet.query;
002
003import java.sql.Timestamp;
004import java.sql.Types;
005import java.time.LocalDateTime;
006import java.util.ArrayList;
007import java.util.Collections;
008import java.util.List;
009import java.util.Map;
010import java.util.Map.Entry;
011import java.util.Objects;
012import java.util.Set;
013import java.util.stream.Collectors;
014
015import org.springframework.jdbc.core.SqlParameterValue;
016import org.springframework.util.Assert;
017import org.springframework.util.StringUtils;
018
019import com.fns.grivet.model.AttributeType;
020
021
022public class DynamicQuery {
023
024    private final List<Constraint> constraints = new ArrayList<>();
025    private final Map<Integer, Integer> attributeToAttributeTypeMap;
026    private final Map<String, Integer> attributeNameToAttributeIdMap;
027    
028    public DynamicQuery(Set<Entry<String, String[]>> requestParams, 
029            Map<Integer, Integer> attributeToAttributeTypeMap, 
030            Map<String, Integer> attributeNameToAttributeIdMap) {
031        String k = null;
032        String[] v = null;
033        if (requestParams != null && !requestParams.isEmpty()) {
034            for (Entry<String, String[]> requestParam: requestParams) {
035                k = requestParam.getKey();
036                v = requestParam.getValue();
037                if (isConstraintKey(k)) {
038                    for (String cs: v) {
039                        if (isConstraintValue(cs)) {
040                            constraints.add(new Constraint(cs.split("\\|")));
041                        }
042                    }
043                }
044            }
045        }
046        this.attributeToAttributeTypeMap = attributeToAttributeTypeMap;
047        this.attributeNameToAttributeIdMap = attributeNameToAttributeIdMap;
048    }
049    
050    public boolean hasConstraints() {
051        return constraints.size() > 0;
052    }
053    
054    List<Constraint> getConstraints() {
055        return Collections.unmodifiableList(constraints);     
056    }
057    
058    public boolean containsAndConjunction() {
059        return constraints
060                .stream()
061                .filter(c -> Objects.equals(c.getConjunction(), Conjunction.AND))
062                .collect(Collectors.toList()).size() >= 1;
063    }
064    
065    public boolean areConjunctionsHomogenous() {
066        boolean result = true;
067        if (constraints.size() >=1) {
068            Conjunction current = null;
069            Conjunction prior = null;
070            for (Constraint c: constraints) {
071                current = c.getConjunction();
072                if (current!= null && prior != null && current != prior) {
073                    result = false;
074                    break;
075                }
076                prior = current;
077            }
078        }
079        return result;
080    }
081    
082    public SqlParameterValue[] asSqlParameterValues() {
083        List<SqlParameterValue> paramValues = new ArrayList<>();
084        Integer attributeId = null;
085        Integer attributeTypeId = null;
086        AttributeType at = null;
087        String attributeName = null;
088        for (Constraint c: constraints) {
089            attributeName = c.getAttributeName();
090            Assert.isTrue(attributeName !=null && !attributeName.isEmpty(), "Attribute name must not be null or empty!");
091            attributeId = attributeNameToAttributeIdMap.get(c.getAttributeName());
092            Assert.notNull(attributeId, String.format("Invalid query constraint! No attribute identifier found for [%s]", attributeName));
093            paramValues.add(new SqlParameterValue(Types.INTEGER, attributeId));
094            attributeTypeId = attributeToAttributeTypeMap.get(attributeId);
095            Assert.notNull(attributeTypeId, String.format("Invalid query constraint! No attribute type identifier found for attribute name [%s]", attributeName));
096            at = AttributeType.fromId(attributeTypeId);
097            Assert.notNull(attributeTypeId, String.format("Invalid query constraint! No matching attribute type found for attribute name [%s]", attributeName));
098            if (c.getOperator().equals(Operator.BETWEEN)) {
099                paramValues.add(getSqlParameterValue(at, c.getOperator(), c.getValues()[0]));
100                paramValues.add(getSqlParameterValue(at, c.getOperator(), c.getValues()[1]));
101            } else if (c.getOperator().equals(Operator.IN)) {
102                for (String value: c.getValues()) {
103                    paramValues.add(getSqlParameterValue(at, c.getOperator(), value));
104                }
105            } else {
106                paramValues.add(getSqlParameterValue(at, c.getOperator(), c.getValues()[0]));
107            }
108        }
109        return paramValues.toArray(new SqlParameterValue[paramValues.size()]);
110    }
111    
112    private boolean isConstraintKey(String key) {
113        boolean result = false;
114        if ((key.equalsIgnoreCase("c") || key.equalsIgnoreCase("constraint"))) {
115            result = true;
116        }
117        return result;
118    }
119    
120    private boolean isConstraintValue(String value) {
121        boolean result = false;
122        if (value != null && StringUtils.countOccurrencesOf(value,"|") >= 2) {
123            result = true;
124        }
125        return result;
126    }
127    
128    protected SqlParameterValue getSqlParameterValue(AttributeType at, Operator op, String value) {
129        SqlParameterValue result = null;
130        switch (at) {
131            case BIG_INTEGER:
132                result = new SqlParameterValue(at.getSqlType(), Long.valueOf(value));
133                break;
134            case DATETIME:
135                result = new SqlParameterValue(at.getSqlType(), Timestamp.valueOf(LocalDateTime.parse(value)));
136                break;
137            case DECIMAL:
138                result = new SqlParameterValue(at.getSqlType(), Double.valueOf(value));
139                break;
140            case INTEGER:
141                result = new SqlParameterValue(at.getSqlType(), Integer.valueOf(value));
142                break;
143            case VARCHAR:
144                result = new SqlParameterValue(at.getSqlType(), getWildcardedValue(op, value));
145                break;
146            case TEXT:
147                result = new SqlParameterValue(at.getSqlType(), getWildcardedValue(op, value));
148                break;
149            case JSON_BLOB:
150                result = new SqlParameterValue(at.getSqlType(), getWildcardedValue(op, value));
151                break;
152        }
153        return result;
154    }
155    
156    protected String getWildcardedValue(Operator op, String value) {
157        String result = null;
158        switch (op) {
159            case STARTS_WITH:
160                result = String.format("%s%s", value, "%");
161                break;
162            case ENDS_WITH:
163                result = String.format("%s%s", "%", value);
164                break;
165            case CONTAINS:
166                result = String.format("%s%s%s", "%", value, "%");
167                break;
168            default:
169                result = value;
170                break;
171        }
172        return result;
173    }
174    
175}
