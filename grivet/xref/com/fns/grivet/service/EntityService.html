55      @Transactional
56      public void create(String type, JSONObject payload) {
57          com.fns.grivet.model.Class c = classRepository.findByName(type);
58          Assert.notNull(c, String.format("Type [%s] is not registered!", type));
59          if (c.isValidatable()) {
60              ProcessingReport report = schemaValidator.validate(type, payload);
61              Assert.isTrue(report.isSuccess(), String.format("Cannot store [%s]! Type does not conform to JSON Schema definition.\n %s", type, report.toString()));
62          }
63          Set<String> keys = payload.keySet();
64          Assert.notEmpty(keys, String.format("Type [%s] must declare at least one attribute in create request!", type));
65          Object val = null;
66          Attribute a = null;
67          ClassAttribute ca = null;
68          AttributeType at = null;
69          Long eid = entityRepository.id(c.getId());
70          // TODO create an expiring cache of attribute names for faster lookups (friendlier database use)
71          for (String key: keys) {
72              val = payload.get(key);
73              a = attributeRepository.findByName(key);
74              Assert.notNull(a, String.format("Attribute [%s] is not registered!", key));
75              ca = classAttributeRepository.findByCidAndAid(c.getId(), a.getId());
76              Assert.notNull(ca, String.format("[%s] is not a valid attribute of [%s]", key, type));
77              at = attributeTypeRepository.findOne(ca.getTid());
78              Assert.notNull(at, String.format("Attribute type [%s] is not supported!", at));
79              entityRepository.save(eid, a, at, val);
80          }
81      }
82      
83      @Transactional(readOnly=true)
84      public String get(String type, LocalDateTime createdTimeStart, LocalDateTime createdTimeEnd, Set<Entry<String, String[]>> requestParameters) throws JsonProcessingException {
85          com.fns.grivet.model.Class c = classRepository.findByName(type);
86          Assert.notNull(c, "Type [%s] is not registered!");
87          List<ClassAttribute> cas = classAttributeRepository.findByCid(c.getId());
88          List<Attribute> attributes = Lists.newArrayList(attributeRepository.findAll());
89          Map<Integer, Integer> attributeToAttributeTypeMap = cas.stream().collect(Collectors.toMap(ClassAttribute::getAid, ClassAttribute::getTid));
90          Map<String, Integer> attributeNameToAttributeIdMap = attributes.stream().collect(Collectors.toMap(Attribute::getName, Attribute::getId));
91          DynamicQuery query = new DynamicQuery(requestParameters, attributeToAttributeTypeMap, attributeNameToAttributeIdMap);
92          List<EntityAttributeValue> rows = null;
93          if (query.hasConstraints()) {
94              rows = entityRepository.executeDynamicQuery(c.getId(), query);
95          } else {
96              rows = entityRepository.find(c.getId(), createdTimeStart, createdTimeEnd);
97          }
98          return mapRows(attributeToAttributeTypeMap, rows);
99      }
100 
101     protected String mapRows(Map<Integer, Integer> attributeToAttributeTypeMap, List<EntityAttributeValue> rows) {
102         JSONArray jsonArray = new JSONArray();
103         Long current = null; 
104         Long previous = null;
105         JSONObject jsonObject = null;
106         for (EntityAttributeValue row: rows) {
107             current = row.getId();
108             if (current != previous) {
109                 jsonObject = new JSONObject();
110                 jsonArray.put(jsonObject);
111             }
112             Integer tid = attributeToAttributeTypeMap.get(row.getAttributeId());
113             jsonObject.put(row.getAttributeName(), ValueHelper.getValue(attributeTypeRepository.findOne(tid), row.getAttributeValue()));
114             previous = current;
115         }
116         return jsonArray.toString();
117     }
