170     @Test
171     public void testRegisterAndLinkAndStoreAndGetTypeHappyPath() {
172         registerTestType();
173         Resource r = resolver.getResource("classpath:TestTypeSchema.json");
174         try {
175 	        String schema = IOUtils.toString(r.getInputStream());
176 	        given().contentType("application/json").request().body(schema).then().expect().statusCode(equalTo(200)).when()
177 	                .post("/schema");
178 	        r = resolver.getResource("classpath:TestTypeData.json");
179 	        String type = IOUtils.toString(r.getInputStream());
180 	        given().contentType("application/json").request().body(type).then().expect().statusCode(equalTo(201)).when()
181 	                .post("/store/TestType");
182 	        
183 	        // GET (default)
184 	        Response response = given().contentType("application/json").request().then().expect().statusCode(equalTo(200))
185 	                .when().get("/store/TestType");
186 	        JSONArray result = new JSONArray(response.body().asString());
187 	        JsonAssert.assertJsonEquals(type, result.get(0).toString());
188 	        
189 	        // GET (with equals constraint that is a boolean)
190 	        response = given().param("c", "boolean|equals|false").contentType("application/json").request().then().expect()
191 	                .statusCode(equalTo(200)).when().get("/store/TestType");
192 	        result = new JSONArray(response.body().asString());
193 	        JsonAssert.assertJsonEquals(type, result.get(0).toString());
194 	        
195 	        // GET (with equals constraint that is a varchar)
196 	        response = given().param("c", "varchar|equals|Rush").contentType("application/json").request().then().expect()
197 	                .statusCode(equalTo(200)).when().get("/store/TestType");
198 	        result = new JSONArray(response.body().asString());
199 	        JsonAssert.assertJsonEquals(type, result.get(0).toString());
200 	        
201 	        // GET (with startsWith constraint)
202 	        response = given().param("c", "text|startsWith|Grim-faced").contentType("application/json").request().then()
203 	                .expect().statusCode(equalTo(200)).when().get("/store/TestType");
204 	        result = new JSONArray(response.body().asString());
205 	        JsonAssert.assertJsonEquals(type, result.get(0).toString());
206 	        
207 	        // GET (with lessThanOrEqualTo constraint)
208 	        response = given().param("c", "datetime|lessThanOrEqualTo|2016-01-01T00:00:00Z").contentType("application/json")
209 	                .request().then().expect().statusCode(equalTo(200)).when().get("/store/TestType");
210 	        result = new JSONArray(response.body().asString());
211 	        JsonAssert.assertJsonEquals(type, result.get(0).toString());
212 	        
213 	        // GET (with multiple constraints)
214 	        response = given()
215 	                    .param("c", "datetime|lessThanOrEqualTo|2016-01-01T00:00:00Z")
216 	                    .param("c", "text|endsWith|city")
217 	                .contentType("application/json").request().then().expect().statusCode(equalTo(200)).when()
218 	                .get("/store/TestType");
219 	        result = new JSONArray(response.body().asString());
220 	        JsonAssert.assertJsonEquals(type, result.get(0).toString());
221         } catch (IOException e) {
222         	fail(e.getMessage());
223         }
224         
225         deregisterType();
226     }
